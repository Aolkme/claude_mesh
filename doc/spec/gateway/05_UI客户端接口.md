# 05_UI客户端接口

> 本文档定义 meshgate-ui Qt 图形界面客户端的设计与实现。

## 1. 概述

### 1.1 设计目标

- 跨平台图形界面 (Qt5/Qt6)
- 与服务端解耦，可独立运行
- 实时数据展示
- 友好的交互体验

### 1.2 技术选型

| 组件 | 技术 |
|------|------|
| UI 框架 | Qt 5.15+ / Qt 6.x |
| 构建 | CMake / qmake |
| 通信 | QLocalSocket |
| 数据格式 | JSON (QJsonDocument) |

## 3. Socket 客户端

### 3.1 接口定义

```cpp
// SocketClient.h
#ifndef SOCKETCLIENT_H
#define SOCKETCLIENT_H

#include <QObject>
#include <QLocalSocket>
#include <QJsonObject>
#include <QJsonArray>

class SocketClient : public QObject
{
    Q_OBJECT

public:
    explicit SocketClient(QObject *parent = nullptr);
    ~SocketClient();
    
    // 连接/断开
    bool connectToServer(const QString &socketPath);
    void disconnectFromServer();
    bool isConnected() const;
    
    // 发送命令
    int sendCommand(const QString &cmd, const QJsonObject &params = QJsonObject());
    void sendCommand(int id, const QString &cmd, const QJsonObject &params = QJsonObject());
    
    // 订阅事件
    void subscribe(const QStringList &events);
    void unsubscribe(const QStringList &events);
    
signals:
    void connected();
    void disconnected();
    void connectionError(const QString &error);
    
    void responseReceived(int id, const QJsonObject &data);
    void errorReceived(int id, int code, const QString &message);
    
    void eventReceived(const QString &event, const QJsonObject &data);
    
private slots:
    void onReadyRead();
    void onConnected();
    void onDisconnected();
    void onError(QLocalSocket::LocalSocketError error);
    
private:
    QLocalSocket *m_socket;
    QByteArray m_buffer;
    int m_nextId;
    bool parseMessage(const QByteArray &data);
};

#endif // SOCKETCLIENT_H
```

### 3.2 实现

```cpp
// SocketClient.cpp
#include "SocketClient.h"
#include <QJsonDocument>
#include <QDataStream>

SocketClient::SocketClient(QObject *parent)
    : QObject(parent)
    , m_socket(new QLocalSocket(this))
    , m_nextId(1)
{
    connect(m_socket, &QLocalSocket::readyRead, this, &SocketClient::onReadyRead);
    connect(m_socket, &QLocalSocket::connected, this, &SocketClient::onConnected);
    connect(m_socket, &QLocalSocket::disconnected, this, &SocketClient::onDisconnected);
    connect(m_socket, &QLocalSocket::errorOccurred, this, &SocketClient::onError);
}

SocketClient::~SocketClient()
{
    disconnectFromServer();
}

bool SocketClient::connectToServer(const QString &socketPath)
{
    m_socket->connectToServer(socketPath);
    return m_socket->waitForConnected(5000);
}

void SocketClient::disconnectFromServer()
{
    if (m_socket->state() == QLocalSocket::ConnectedState) {
        m_socket->disconnectFromServer();
    }
}

bool SocketClient::isConnected() const
{
    return m_socket->state() == QLocalSocket::ConnectedState;
}

int SocketClient::sendCommand(const QString &cmd, const QJsonObject &params)
{
    int id = m_nextId++;
    sendCommand(id, cmd, params);
    return id;
}

void SocketClient::sendCommand(int id, const QString &cmd, const QJsonObject &params)
{
    QJsonObject request;
    request["id"] = id;
    request["cmd"] = cmd;
    if (!params.isEmpty()) {
        request["params"] = params;
    }
    
    QJsonDocument doc(request);
    QByteArray data = doc.toJson(QJsonDocument::Compact) + "\n";
    
    m_socket->write(data);
    m_socket->flush();
}

void SocketClient::subscribe(const QStringList &events)
{
    QJsonObject params;
    QJsonArray arr;
    for (const QString &e : events) {
        arr.append(e);
    }
    params["events"] = arr;
    sendCommand("subscribe", params);
}

void SocketClient::unsubscribe(const QStringList &events)
{
    QJsonObject params;
    QJsonArray arr;
    for (const QString &e : events) {
        arr.append(e);
    }
    params["events"] = arr;
    sendCommand("unsubscribe", params);
}

void SocketClient::onReadyRead()
{
    m_buffer.append(m_socket->readAll());
    
    while (true) {
        int newlinePos = m_buffer.indexOf('\n');
        if (newlinePos < 0) break;
        
        QByteArray line = m_buffer.left(newlinePos);
        m_buffer.remove(0, newlinePos + 1);
        
        parseMessage(line);
    }
}

bool SocketClient::parseMessage(const QByteArray &data)
{
    QJsonParseError error;
    QJsonDocument doc = QJsonDocument::fromJson(data, &error);
    
    if (error.error != QJsonParseError::NoError) {
        emit connectionError("JSON parse error: " + error.errorString());
        return false;
    }
    
    QJsonObject obj = doc.object();
    
    // 响应
    if (obj.contains("id")) {
        int id = obj["id"].toInt();
        QString status = obj["status"].toString();
        
        if (status == "ok") {
            emit responseReceived(id, obj["data"].toObject());
        } else if (status == "error") {
            QJsonObject err = obj["error"].toObject();
            emit errorReceived(id, err["code"].toInt(), err["message"].toString());
        }
    }
    // 事件
    else if (obj.contains("event")) {
        QString event = obj["event"].toString();
        emit eventReceived(event, obj["data"].toObject());
    }
    
    return true;
}

void SocketClient::onConnected()
{
    m_buffer.clear();
    emit connected();
}

void SocketClient::onDisconnected()
{
    emit disconnected();
}

void SocketClient::onError(QLocalSocket::LocalSocketError error)
{
    Q_UNUSED(error);
    emit connectionError(m_socket->errorString());
}
```

```cpp
#ifndef NODEMODEL_H
#define NODEMODEL_H

#include <QAbstractTableModel>
#include <QList>

struct Node {
    QString nodeId;
    QString longName;
    QString shortName;
    int hwModel;
    int batteryLevel;
    double latitude;
    double longitude;
    double snr;
    qint64 lastHeard;

class NodeModel : public QAbstractTableModel
{
    Q_OBJECT

public:
    enum Column {
        ColNodeId = 0,
        ColLongName,
        ColShortName,
        ColBattery,
        ColPosition,
        ColSnr,
        ColLastHeard,
        ColCount
    };

    explicit NodeModel(QObject *parent = nullptr);
    
    int rowCount(const QModelIndex &parent = QModelIndex()) const override;
    int columnCount(const QModelIndex &parent = QModelIndex()) const override;
    QVariant data(const QModelIndex &index, int role = Qt::DisplayRole) const override;
    QVariant headerData(int section, Qt::Orientation orientation, int role) const override;
    
    void updateNode(const QJsonObject &obj);
    void removeNode(const QString &nodeId);
    void clear();
    
    Node* getNode(const QString &nodeId);
    Node* getNode(int row);
    
    int nodeCount() const { return m_nodes.size(); }

private:
    QList<Node> m_nodes;
    QHash<QString, int> m_nodeIndex;
};
```

### 4.2 节点模型实现

```cpp
// NodeModel.cpp
#include "NodeModel.h"
#include <QDateTime>

NodeModel::NodeModel(QObject *parent)
    : QAbstractTableModel(parent)
{
}

int NodeModel::rowCount(const QModelIndex &parent) const
{
    if (parent.isValid()) return 0;
    return m_nodes.size();
}

int NodeModel::columnCount(const QModelIndex &parent) const
{
    if (parent.isValid()) return 0;
    return ColCount;
}

QVariant NodeModel::data(const QModelIndex &index, int role) const
{
    if (!index.isValid() || index.row() >= m_nodes.size())
        return QVariant();
    
    const Node &node = m_nodes[index.row()];
    
    if (role == Qt::DisplayRole) {
        switch (index.column()) {
            case ColNodeId:
                return node.nodeId;
            case ColLongName:
                return node.longName;
            case ColShortName:
                return node.shortName;
            case ColBattery:
                return node.batteryLevel > 0 ? QString("%1%").arg(node.batteryLevel) : "-";
            case ColPosition:
                if (node.latitude != 0 || node.longitude != 0) {
                    return QString("%1, %2").arg(node.latitude, 0, 'f', 4)
                                           .arg(node.longitude, 0, 'f', 4);
                }
                return "-";
            case ColSnr:
                return QString("%1 dB").arg(node.snr, 0, 'f', 1);
            case ColLastHeard: {
                qint64 secs = QDateTime::currentSecsSinceEpoch() - node.lastHeard;
                if (secs < 60) return QString("%1s ago").arg(secs);
                if (secs < 3600) return QString("%1m ago").arg(secs / 60);
                return QString("%1h ago").arg(secs / 3600);
            }
            default:
                return QVariant();
        }
    }
    
    if (role == Qt::TextAlignmentRole) {
        switch (index.column()) {
            case ColBattery:
            case ColSnr:
                return Qt::AlignCenter;
            default:
                return Qt::AlignLeft | Qt::AlignVCenter;
        }
    }
    
    return QVariant();
}

QVariant NodeModel::headerData(int section, Qt::Orientation orientation, int role) const
{
    if (role != Qt::DisplayRole || orientation != Qt::Horizontal)
        return QVariant();
    
    switch (section) {
        case ColNodeId:    return tr("Node ID");
        case ColLongName:  return tr("Long Name");
        case ColShortName: return tr("Short");
        case ColBattery:   return tr("Battery");
        case ColPosition:  return tr("Position");
        case ColSnr:       return tr("SNR");
        case ColLastHeard: return tr("Last Heard");
        default:           return QVariant();
    }
}

void NodeModel::updateNode(const QJsonObject &obj)
{
    QString nodeId = obj["node_id"].toString();
    
    if (m_nodeIndex.contains(nodeId)) {
        // 更新已有节点
        int row = m_nodeIndex[nodeId];
        Node &node = m_nodes[row];
        
        if (obj.contains("long_name")) node.longName = obj["long_name"].toString();
        if (obj.contains("short_name")) node.shortName = obj["short_name"].toString();
        if (obj.contains("battery_level")) node.batteryLevel = obj["battery_level"].toInt();
        if (obj.contains("latitude")) node.latitude = obj["latitude"].toDouble();
        if (obj.contains("longitude")) node.longitude = obj["longitude"].toDouble();
        if (obj.contains("snr")) node.snr = obj["snr"].toDouble();
        node.lastHeard = QDateTime::currentSecsSinceEpoch();
        
        emit dataChanged(index(row, 0), index(row, ColCount - 1));
    } else {
        // 添加新节点
        beginInsertRows(QModelIndex(), m_nodes.size(), m_nodes.size());
        
        Node node;
        node.nodeId = nodeId;
        node.longName = obj["long_name"].toString();
        node.shortName = obj["short_name"].toString();
        node.hwModel = obj["hw_model"].toInt();
        node.batteryLevel = obj["battery_level"].toInt();
        node.latitude = obj["latitude"].toDouble();
        node.longitude = obj["longitude"].toDouble();
        node.snr = obj["snr"].toDouble();
        node.lastHeard = QDateTime::currentSecsSinceEpoch();
        node.isValid = true;
        
        m_nodeIndex[nodeId] = m_nodes.size();
        m_nodes.append(node);
        
        endInsertRows();
    }
}

void NodeModel::removeNode(const QString &nodeId)
{
    if (!m_nodeIndex.contains(nodeId)) return;
    
    int row = m_nodeIndex[nodeId];
    beginRemoveRows(QModelIndex(), row, row);
    m_nodes.removeAt(row);
    m_nodeIndex.remove(nodeId);
    
    // 重建索引
    m_nodeIndex.clear();
    for (int i = 0; i < m_nodes.size(); i++) {
        m_nodeIndex[m_nodes[i].nodeId] = i;
    }
    
    endRemoveRows();
}

void NodeModel::clear()
{
    beginResetModel();
    m_nodes.clear();
    m_nodeIndex.clear();
    endResetModel();
}

Node* NodeModel::getNode(const QString &nodeId)
{
    if (m_nodeIndex.contains(nodeId)) {
        return &m_nodes[m_nodeIndex[nodeId]];
    }
    return nullptr;
}

Node* NodeModel::getNode(int row)
{
    if (row >= 0 && row < m_nodes.size()) {
        return &m_nodes[row];
    }
    return nullptr;
}
```

```cpp
// MainWindow.cpp
#include "MainWindow.h"
#include "ConnectionWidget.h"
#include "NodeListView.h"
#include "MessageView.h"
#include "SendPanel.h"
#include "LogView.h"

#include <QSplitter>
#include <QTabWidget>
#include <QToolBar>
#include <QStatusBar>
#include <QMessageBox>
#include <QSettings>
#include <QApplication>

MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent)
    , m_client(new SocketClient(this))
    , m_nodeModel(new NodeModel(this))
    , m_socketPath("/run/meshgateway.sock")
    , m_pendingCommandId(0)
{
    setupUi();
    setupConnections();
    updateStatusBar();
    
    // 加载设置
    QSettings settings;
    m_socketPath = settings.value("socketPath", m_socketPath).toString();
    
    // 自动连接
    if (settings.value("autoConnect", true).toBool()) {
        onConnectClicked();
    }
}

MainWindow::~MainWindow()
{
    QSettings settings;
    settings.setValue("socketPath", m_socketPath);
}

void MainWindow::setupUi()
{
    setWindowTitle(tr("Meshtastic Gateway"));
    resize(1200, 800);
    
    // 工具栏
    QToolBar *toolbar = addToolBar(tr("Main"));
    toolbar->addAction(tr("Connect"), this, &MainWindow::onConnectClicked);
    toolbar->addAction(tr("Disconnect"), this, &MainWindow::onDisconnectClicked);
    toolbar->addSeparator();
    toolbar->addAction(tr("Refresh"), this, &MainWindow::onRefreshClicked);
    
    // 中央部件
    QSplitter *splitter = new QSplitter(Qt::Horizontal, this);
    
    // 左侧: 连接状态
    m_connectionWidget = new ConnectionWidget(this);
    splitter->addWidget(m_connectionWidget);
    
    // 右侧: 标签页
    QTabWidget *tabWidget = new QTabWidget(this);
    
    m_nodeListView = new NodeListView(m_nodeModel, this);
    tabWidget->addTab(m_nodeListView, tr("Nodes"));
    
    m_messageView = new MessageView(this);
    tabWidget->addTab(m_messageView, tr("Messages"));
    
    m_logView = new LogView(this);
    tabWidget->addTab(m_logView, tr("Log"));
    
    splitter->addWidget(tabWidget);
    splitter->setSizes({200, 1000});
    
    setCentralWidget(splitter);
    
    // 发送面板
    m_sendPanel = new SendPanel(m_nodeModel, this);
    m_sendPanel->setAllowedAreas(Qt::BottomDockWidgetArea);
    addDockWidget(Qt::BottomDockWidgetArea, m_sendPanel);
    
    // 状态栏
    m_statusLabel = new QLabel(tr("Disconnected"), this);
    m_nodeCountLabel = new QLabel(tr("Nodes: 0"), this);
    m_serialStatusLabel = new QLabel(tr("Serial: -"), this);
    
    statusBar()->addWidget(m_statusLabel, 1);
    statusBar()->addPermanentWidget(m_nodeCountLabel);
    statusBar()->addPermanentWidget(m_serialStatusLabel);
}

void MainWindow::setupConnections()
{
    // 客户端信号
    connect(m_client, &SocketClient::connected, this, &MainWindow::onClientConnected);
    connect(m_client, &SocketClient::disconnected, this, &MainWindow::onClientDisconnected);
    connect(m_client, &SocketClient::connectionError, this, &MainWindow::onClientError);
    connect(m_client, &SocketClient::responseReceived, this, &MainWindow::onResponseReceived);
    connect(m_client, &SocketClient::errorReceived, this, &MainWindow::onErrorReceived);
    connect(m_client, &SocketClient::eventReceived, this, &MainWindow::onEventReceived);
    
    // 发送面板信号
    connect(m_sendPanel, &SendPanel::sendRequested, this, &MainWindow::onSendRequested);
    
    // 节点选择信号
    connect(m_nodeListView, &NodeListView::nodeSelected, m_sendPanel, &SendPanel::setTargetNode);
}

void MainWindow::onConnectClicked()
{
    m_statusLabel->setText(tr("Connecting..."));
    QApplication::setOverrideCursor(Qt::WaitCursor);
    
    if (!m_client->connectToServer(m_socketPath)) {
        QApplication::restoreOverrideCursor();
        QMessageBox::warning(this, tr("Connection Error"),
                             tr("Failed to connect to gateway service at %1").arg(m_socketPath));
        m_statusLabel->setText(tr("Disconnected"));
    }
}

void MainWindow::onDisconnectClicked()
{
    m_client->disconnectFromServer();
}

void MainWindow::onRefreshClicked()
{
    if (m_client->isConnected()) {
        m_client->sendCommand("get_nodes");
    }
}

void MainWindow::onClientConnected()
{
    QApplication::restoreOverrideCursor();
    m_statusLabel->setText(tr("Connected"));
    
    // 订阅事件
    m_client->subscribe({"node_updated", "node_expired", "packet_received", 
                         "telemetry_received", "serial_state_changed"});
    
    // 获取初始数据
    m_client->sendCommand("get_nodes");
    m_client->sendCommand("get_status");
}

void MainWindow::onClientDisconnected()
{
    m_statusLabel->setText(tr("Disconnected"));
    m_nodeModel->clear();
    updateStatusBar();
}

void MainWindow::onClientError(const QString &error)
{
    m_logView->appendLog(tr("Error: %1").arg(error));
}

void MainWindow::onResponseReceived(int id, const QJsonObject &data)
{
    Q_UNUSED(id);
    
    // 处理节点列表响应
    if (data.contains("nodes")) {
        QJsonArray nodes = data["nodes"].toArray();
        for (const QJsonValue &v : nodes) {
            m_nodeModel->updateNode(v.toObject());
        }
        updateStatusBar();
    }
    
    // 处理状态响应
    if (data.contains("serial_state")) {
        QString state = data["serial_state"].toString();
        m_serialStatusLabel->setText(tr("Serial: %1").arg(state));
    }
}

void MainWindow::onErrorReceived(int id, int code, const QString &message)
{
    Q_UNUSED(id);
    Q_UNUSED(code);
    m_logView->appendLog(tr("Error: %1").arg(message));
}

void MainWindow::onEventReceived(const QString &event, const QJsonObject &data)
{
    m_logView->appendLog(tr("Event: %1").arg(event));
    
    if (event == "node_updated") {
        m_nodeModel->updateNode(data["node"].toObject());
        updateStatusBar();
    } else if (event == "node_expired") {
        m_nodeModel->removeNode(data["node_id"].toString());
        updateStatusBar();
    } else if (event == "packet_received") {
        m_messageView->addPacket(data);
    } else if (event == "serial_state_changed") {
        m_serialStatusLabel->setText(tr("Serial: %1").arg(data["state"].toString()));
    }
}

void MainWindow::onSendRequested(const QJsonObject &params)
{
    if (!m_client->isConnected()) {
        QMessageBox::warning(this, tr("Not Connected"), 
                             tr("Please connect to gateway first"));
        return;
    }
    
    QString cmd = params["cmd"].toString();
    QJsonObject cmdParams = params["params"].toObject();
    m_client->sendCommand(cmd, cmdParams);
}

void MainWindow::updateStatusBar()
{
    m_nodeCountLabel->setText(tr("Nodes: %1").arg(m_nodeModel->nodeCount()));
}
```

## 7. 编译配置

### 7.1 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.16)
project(meshgate-ui VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Widgets Network)

set(SOURCES
    src/main.cpp
    src/MainWindow.cpp
    src/widgets/ConnectionWidget.cpp
    src/widgets/NodeListView.cpp
    src/widgets/MessageView.cpp
    src/widgets/SendPanel.cpp
    src/widgets/LogView.cpp
    src/models/NodeModel.cpp
    src/models/MessageModel.cpp
    src/client/SocketClient.cpp
)

set(HEADERS
    src/MainWindow.h
    src/widgets/ConnectionWidget.h
    src/widgets/NodeListView.h
    src/widgets/MessageView.h
    src/widgets/SendPanel.h
    src/widgets/LogView.h
    src/models/NodeModel.h
    src/models/MessageModel.h
    src/client/SocketClient.h
)

add_executable(meshgate-ui ${SOURCES} ${HEADERS})

target_include_directories(meshgate-ui PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_link_libraries(meshgate-ui PRIVATE
    Qt6::Core
    Qt6::Widgets
    Qt6::Network
)

install(TARGETS meshgate-ui DESTINATION bin)
```

### 7.2 运行

```bash
# 编译
mkdir build && cd build
cmake ..
make

# 运行
./meshgate-ui
```
