## 3. PRIVATE_CONFIG_APP (287)

### 3.1 概述

PRIVATE_CONFIG_APP 是自定义的私有配置端口，用于：

1. **设备入网** - 空设备接收公司密钥，成为入网设备
2. **更换管理员** - 使用 Ed25519 签名验证，将管理员权限转移给新网关
3. **同步唤醒配置** - 配置设备的同步唤醒策略
4. **设备信息标签** - 读取/设置设备的信息标签

**关键架构说明：**
- 网关节点**物理上是管理员**（admin_key[0] 存储网关公钥）
- 网关发送 Admin 命令时，**使用自己的节点私钥**，走原生 PKI 路径
- 无需伪造 from 字段，无需手动管理 Session Passkey

```protobuf
package temeshtastic;

/* 同步唤醒策略 */
enum WakeupStrategy {
    STRATEGY_FIXED = 0;      /* 固定时间唤醒 */
    STRATEGY_SCHEDULED = 1;  /* 按时间表唤醒 */
}

/* 固定唤醒配置 (字段号从3开始，与固件一致) */
message SyncWakeupConfig_FixedWakeup {
    uint32 interval_min = 3;     /* 间隔分钟数 */
    uint32 align_minute = 4;     /* 对齐分钟数 (0-59) */
    uint32 offset_sec = 5;       /* 偏移秒数 */
}

/* 时间段配置 */
message SyncWakeupConfig_TimeSlot {
    uint32 start_hour = 1;       /* 开始小时 */
    uint32 end_hour = 2;         /* 结束小时 */
    uint32 interval_min = 3;     /* 间隔分钟数 */
    uint32 align_minute = 4;     /* 对齐分钟数 */
}

/* 按时间表唤醒配置 */
message SyncWakeupConfig_ScheduledWakeup {
    uint32 offset_sec = 1;       /* 偏移秒数 */
    repeated SyncWakeupConfig_TimeSlot time_slots = 6;
}

/* 同步唤醒配置 */
message SyncWakeupConfig {
    bool enabled = 1;
    WakeupStrategy strategy = 2;
    SyncWakeupConfig_FixedWakeup fixed_wakeup = 3;
    SyncWakeupConfig_ScheduledWakeup scheduled_wakeup = 4;
}

/* 私有配置 */
message PrivateConfig {
    uint32 privateVersion = 1;
    string deviceName = 2;
    bool isAdminKeySet = 3;
    SyncWakeupConfig sync_wakeup = 4;
}

/* 设备信息标签 */
message DeviceLabels_InfoLabel {
    uint32 id = 1;
    string key = 2;
    string value = 3;
}

message DeviceLabels {
    repeated DeviceLabels_InfoLabel info_labels = 1;
}

/* 公司配置 (存储在设备上) */
message CompanyConfig {
    bytes company_public_key = 1;       /* 公司公钥 (32字节) */
    bool is_enrolled = 2;               /* 是否已入网 */
    uint64 last_change_timestamp = 3;   /* 最后变更时间戳 */
}

/* 首次入网请求 */
message PrivateConfigPacket_SetCompanyKeyRequest {
    bytes company_public_key = 1;       /* 公司公钥 (32字节) */
    bytes gateway_public_key = 2;       /* 网关公钥 (32字节) */
    uint32 gateway_node_id = 3;         /* 网关节点号 */
    uint64 timestamp = 4;               /* 时间戳 */
}

/* 更换管理员请求 */
message PrivateConfigPacket_ChangeAdminRequest {
    bytes new_gateway_public_key = 1;   /* 新网关公钥 (32字节) */
    uint32 new_gateway_node_id = 2;     /* 新网关节点号 */
    uint64 timestamp = 3;               /* 时间戳 */

}

/* 设置同步唤醒配置 */
message PrivateConfigPacket_SetSyncWakeupConfig {
    SyncWakeupConfig config = 1;
}

/* 请求消息 */
message PrivateConfigPacket_Request {
    uint32 status = 1;                  /* 状态码 */
    oneof payload {
        bool get_config = 2;            /* 获取完整配置 */
        bool get_info_labels = 3;       /* 获取信息标签 */
        bool get_sync_wakeup = 4;       /* 获取唤醒配置 */
        PrivateConfigPacket_SetSyncWakeupConfig set_sync_wakeup_config = 8;
        PrivateConfigPacket_SetCompanyKeyRequest set_company_key = 9;   /* 首次入网 */
        PrivateConfigPacket_ChangeAdminRequest change_admin = 10;       /* 更换管理员 */
    }
}

/* 响应消息 */
message PrivateConfigPacket_Response {
    oneof payload {
        PrivateConfig private_config = 1;
        DeviceLabels device_labels = 2;
        SyncWakeupConfig sync_wakeup_config = 3;
        CompanyConfig company_config = 4;
    }
}

/* 完整包结构 */
message PrivateConfigPacket {
    oneof packet_type {
        PrivateConfigPacket_Request request = 1;
        PrivateConfigPacket_Response response = 2;
    }
}
```

## 3.5 Ed25519 签名验证

### 3.5.1 签名数据构造

更换管理员时，需要使用**公司私钥**对以下数据进行 Ed25519 签名：

```
签名数据 (44字节) = new_gateway_public_key (32字节) + timestamp (8字节) + device_node_id (4字节)

┌─────────────────────────────────┬────────────────┬──────────────┐
│ new_gateway_public_key          │ timestamp      │ device_node_id│
│ (32 bytes)                      │ (8 bytes)      │ (4 bytes)     │
│ 小端字节序                       │ 小端字节序      │ 小端字节序     │
└─────────────────────────────────┴────────────────┴──────────────┘
```

**构造代码示例：**

```c
/* 构造签名数据 */
void build_signature_data(
    const uint8_t new_gateway_pubkey[32],
    uint64_t timestamp,
    uint32_t device_node_id,
    uint8_t out[44])
{
    memcpy(out, new_gateway_pubkey, 32);
    memcpy(out + 32, &timestamp, 8);      /* 小端 */
    memcpy(out + 40, &device_node_id, 4); /* 小端 */
}

/* Ed25519 签名 (需要加密库支持) */
bool sign_change_admin(
    const uint8_t company_private_key[32],
    const uint8_t new_gateway_pubkey[32],
    uint64_t timestamp,
    uint32_t device_node_id,
    uint8_t signature_out[64])
{
    uint8_t data[44];
    build_signature_data(new_gateway_pubkey, timestamp, device_node_id, data);
    
    /* 使用 Ed25519 私钥签名 */
    /* crypto_sign(signature_out, NULL, data, 44, company_private_key); */
    
    return true;
}
```

### 3.5.2 设备端验证流程

设备收到 `change_admin` 请求后：

1. 从 `CompanyConfig.company_public_key` 获取公司公钥
2. 构造签名数据：`new_gateway_public_key + timestamp + self.node_id`
3. 使用公司公钥验证 Ed25519 签名
4. 验证通过后，更新 `admin_key[0] = new_gateway_public_key`

## 3.6 三阶段流程

### 3.6.1 阶段一：设备入网

```
┌─────────────┐                    ┌─────────────┐
│   网关      │                    │   空设备    │
│ (Gateway)   │                    │ (is_enrolled=false) │
└──────┬──────┘                    └──────┬──────┘
       │                                  │
       │  SetCompanyKeyRequest            │
       │  (company_pubkey, gw_pubkey,     │
       │   gw_node_id, timestamp)         │
       │ ─────────────────────────────────>│
       │                                  │
       │                        ┌─────────┴─────────┐
       │                        │ 1. 无需验证签名    │
       │                        │ 2. 存储 company   │
       │                        │    public_key     │
       │                        │ 3. admin_key[0] = │
       │                        │    gateway_pubkey │
       │                        │ 4. is_enrolled=true│
       │                        └─────────┬─────────┘
       │                                  │
       │         Response (success)       │
       │ <─────────────────────────────────│
       │                                  │
       │         设备入网完成              │
```

**关键点：**
- 空设备（`is_enrolled = false`）接受任何 `SetCompanyKey` 请求
- **无需签名验证** - 这是设计核心
- 入网后，网关即成为该设备的管理员

### 3.6.2 阶段二：正常管理

```
┌─────────────┐                    ┌─────────────┐
│   网关      │                    │   设备      │
│ (admin_key) │                    │ (is_enrolled=true) │
└──────┬──────┘                    └──────┬──────┘
       │                                  │
       │  AdminMessage (标准Admin命令)     │
       │  使用网关自己的节点私钥签名        │

       │                        ┌─────────┴─────────┐
       │                        │ 原生 AdminModule: │
       │                        │ 1. 验证 from 是否 │
       │                        │    匹配 admin_key │
       │                        │ 2. 执行命令       │
       │                        └─────────┬─────────┘
       │                                  │
       │         Response                 │
       │ <─────────────────────────────────│
```

**关键点：**
- 网关**物理上是管理员**，admin_key[0] 存储网关公钥
- 发送 Admin 命令时，使用**自己的节点私钥**
- 走**原生 PKI 路径**，无需伪造 from 字段
- Session Passkey 由 Meshtastic 原生 AdminModule 自动处理

### 3.7 实现

```c
// private_config.c
#include "private_config.h"
#include <string.h>
#include <time.h>

const char* wakeup_strategy_name(wakeup_strategy_t strategy) {
    switch (strategy) {
        case WAKEUP_STRATEGY_FIXED:     return "FIXED";
        case WAKEUP_STRATEGY_SCHEDULED: return "SCHEDULED";
        default: return "UNKNOWN";
    }
}

/* 
 * Protobuf 编码辅助函数
 */

/* 编码 varint */
static int encode_varint(uint8_t *buf, uint32_t value) {
    int len = 0;
    do {
        buf[len] = (value & 0x7F) | ((value > 0x7F) ? 0x80 : 0);
        len++;
        value >>= 7;
    } while (value);
    return len;
}

/* 编码 varint (64位) */
static int encode_varint64(uint8_t *buf, uint64_t value) {
    int len = 0;
    do {
        buf[len] = (value & 0x7F) | ((value > 0x7F) ? 0x80 : 0);
        len++;
        value >>= 7;
    } while (value);
    return len;
}

/* 编码 tag */
static int encode_tag(uint8_t *buf, uint32_t field_num, uint32_t wire_type) {
    return encode_varint(buf, (field_num << 3) | wire_type);
}

/* wire types */
#define WIRE_TYPE_VARINT  0
#define WIRE_TYPE_LENGTH  2

/* 编码 bytes 字段 */
static int encode_bytes(uint8_t *buf, uint32_t field_num, 
                        const uint8_t *data, uint16_t len) {
    int pos = encode_tag(buf, field_num, WIRE_TYPE_LENGTH);
    pos += encode_varint(buf + pos, len);
    memcpy(buf + pos, data, len);
    return pos + len;
}

uint16_t private_config_build_get_config(uint8_t *out, uint16_t out_size) {
    if (out == NULL || out_size < 8) return 0;
    
    uint16_t len = 0;
    uint8_t inner[8];
    uint16_t inner_len = 0;
    
    /* request.get_config = true (field 2) */
    inner_len += encode_tag(inner + inner_len, 2, WIRE_TYPE_VARINT);
    inner_len += encode_varint(inner + inner_len, 1);
    
    /* packet_type = request (field 1) */
    len += encode_tag(out + len, 1, WIRE_TYPE_LENGTH);
    len += encode_varint(out + len, inner_len);
    memcpy(out + len, inner, inner_len);
    
    return len + inner_len;
}

uint16_t private_config_build_get_sync_wakeup(uint8_t *out, uint16_t out_size) {
    if (out == NULL || out_size < 16) return 0;
    
    uint16_t len = 0;
    uint8_t inner[8];
    uint16_t inner_len = 0;
    
    /* request.get_sync_wakeup = true (field 4) */
    inner_len += encode_tag(inner + inner_len, 4, WIRE_TYPE_VARINT);
    inner_len += encode_varint(inner + inner_len, 1);
    
    len += encode_tag(out + len, 1, WIRE_TYPE_LENGTH);
    len += encode_varint(out + len, inner_len);
    memcpy(out + len, inner, inner_len);
    
    return len + inner_len;
}

uint16_t private_config_build_set_sync_wakeup(const sync_wakeup_config_t *config,
                                               uint8_t *out, uint16_t out_size) {
    if (config == NULL || out == NULL || out_size < 128) return 0;
    
    uint16_t len = 0;
    
    /* 构建 SyncWakeupConfig 消息 */
    uint8_t cfg_msg[64];
    uint16_t cfg_len = 0;
    
    cfg_len += encode_tag(cfg_msg + cfg_len, 1, WIRE_TYPE_VARINT);
    cfg_len += encode_varint(cfg_msg + cfg_len, config->enabled ? 1 : 0);
    
    cfg_len += encode_tag(cfg_msg + cfg_len, 2, WIRE_TYPE_VARINT);
    cfg_len += encode_varint(cfg_msg + cfg_len, config->strategy);
    
    if (config->strategy == WAKEUP_STRATEGY_FIXED) {
        uint8_t fixed_msg[32];
        uint16_t fixed_len = 0;
        
        fixed_len += encode_tag(fixed_msg + fixed_len, 3, WIRE_TYPE_VARINT);
        fixed_len += encode_varint(fixed_msg + fixed_len, config->fixed_wakeup.interval_min);
        
        fixed_len += encode_tag(fixed_msg + fixed_len, 4, WIRE_TYPE_VARINT);
        fixed_len += encode_varint(fixed_msg + fixed_len, config->fixed_wakeup.align_minute);
        
        fixed_len += encode_tag(fixed_msg + fixed_len, 5, WIRE_TYPE_VARINT);
        fixed_len += encode_varint(fixed_msg + fixed_len, config->fixed_wakeup.offset_sec);
        
        cfg_len += encode_tag(cfg_msg + cfg_len, 3, WIRE_TYPE_LENGTH);
        cfg_len += encode_varint(cfg_msg + cfg_len, fixed_len);
        memcpy(cfg_msg + cfg_len, fixed_msg, fixed_len);
        cfg_len += fixed_len;
    }
    
    /* 构建 SetSyncWakeupConfig (field 1 = config) */
    uint8_t sswc_msg[64];
    uint16_t sswc_len = 0;
    sswc_len += encode_bytes(sswc_msg, 1, cfg_msg, cfg_len);
    
    /* 构建 request.set_sync_wakeup_config (field 8) */
    uint8_t req_msg[80];
    uint16_t req_len = 0;
    req_len += encode_bytes(req_msg, 8, sswc_msg, sswc_len);
    
    /* 构建最终包 */
    len += encode_tag(out + len, 1, WIRE_TYPE_LENGTH);
    len += encode_varint(out + len, req_len);
    memcpy(out + len, req_msg, req_len);
    
    return len + req_len;
}

uint16_t private_config_build_set_company_key(
    const uint8_t company_pubkey[PUBLIC_KEY_LEN],
    const uint8_t gateway_pubkey[PUBLIC_KEY_LEN],
    uint32_t gateway_node_id,
    uint64_t timestamp,
    uint8_t *out, uint16_t out_size)
{
    if (out == NULL || out_size < 128) return 0;
    
    uint16_t len = 0;
    
    /* 构建 SetCompanyKeyRequest 消息 */
    uint8_t sck_msg[96];
    uint16_t sck_len = 0;
    
    /* company_public_key (field 1) */
    sck_len += encode_bytes(sck_msg + sck_len, 1, company_pubkey, PUBLIC_KEY_LEN);
    
    /* gateway_public_key (field 2) */
    sck_len += encode_bytes(sck_msg + sck_len, 2, gateway_pubkey, PUBLIC_KEY_LEN);
    
    /* gateway_node_id (field 3) */
    sck_len += encode_tag(sck_msg + sck_len, 3, WIRE_TYPE_VARINT);
    sck_len += encode_varint(sck_msg + sck_len, gateway_node_id);
    
    /* timestamp (field 4) */
    sck_len += encode_tag(sck_msg + sck_len, 4, WIRE_TYPE_VARINT);
    sck_len += encode_varint64(sck_msg + sck_len, timestamp);
    
    /* 构建 request.set_company_key (field 9) */
    uint8_t req_msg[100];
    uint16_t req_len = 0;
    req_len += encode_bytes(req_msg, 9, sck_msg, sck_len);
    
    /* 构建最终包 */
    len += encode_tag(out + len, 1, WIRE_TYPE_LENGTH);
    len += encode_varint(out + len, req_len);
    memcpy(out + len, req_msg, req_len);
    
    return len + req_len;
}

uint16_t private_config_build_change_admin(
    const uint8_t new_gateway_pubkey[PUBLIC_KEY_LEN],
    uint32_t new_gateway_node_id,
    uint64_t timestamp,
    const uint8_t signature[SIGNATURE_LEN],
    uint8_t *out, uint16_t out_size)
{
    if (out == NULL || out_size < 200) return 0;
    
    uint16_t len = 0;
    
    /* 构建 ChangeAdminRequest 消息 */
    uint8_t ca_msg[128];
    uint16_t ca_len = 0;
    
    /* new_gateway_public_key (field 1) */
    ca_len += encode_bytes(ca_msg + ca_len, 1, new_gateway_pubkey, PUBLIC_KEY_LEN);
    
    /* new_gateway_node_id (field 2) */
    ca_len += encode_tag(ca_msg + ca_len, 2, WIRE_TYPE_VARINT);
    ca_len += encode_varint(ca_msg + ca_len, new_gateway_node_id);
    
    /* timestamp (field 3) */
    ca_len += encode_tag(ca_msg + ca_len, 3, WIRE_TYPE_VARINT);
    ca_len += encode_varint64(ca_msg + ca_len, timestamp);
    
    /* signature (field 4) */
    ca_len += encode_bytes(ca_msg + ca_len, 4, signature, SIGNATURE_LEN);
    
    /* 构建 request.change_admin (field 10) */
    uint8_t req_msg[132];
    uint16_t req_len = 0;
    req_len += encode_bytes(req_msg, 10, ca_msg, ca_len);
    
    /* 构建最终包 */
    len += encode_tag(out + len, 1, WIRE_TYPE_LENGTH);
    len += encode_varint(out + len, req_len);
    memcpy(out + len, req_msg, req_len);
    
    return len + req_len;
}
```

## 4. WAKEUP_COMM_APP (288)

### 4.1 消息结构

WAKEUP_COMM 使用原始字节格式，不是 Protobuf 消息。

```
┌────────────────────┐
│ Command (1 byte)   │
└────────────────────┘

命令类型:
  0x01 = REQUEST_TELEMETRY    请求遥测数据
  0x02 = SET_CONFIG           下发配置
  0x03 = SYNC_TIME            时间同步
  0x04 = REBOOT               重启设备
```

### 4.2 C 定义

```c
// wakeup_comm.h
#ifndef WAKEUP_COMM_H
#define WAKEUP_COMM_H

#include <stdint.h>

/* 命令类型 */
typedef enum {
    WAKEUP_CMD_REQUEST_TELEMETRY = 0x01,
    WAKEUP_CMD_SET_CONFIG = 0x02,
    WAKEUP_CMD_SYNC_TIME = 0x03,
    WAKEUP_CMD_REBOOT = 0x04,
} wakeup_command_t;

/**
 * 构建唤醒命令 payload
 */
static inline uint16_t wakeup_comm_build(wakeup_command_t cmd, uint8_t *out) {
    if (out == NULL) return 0;
    out[0] = (uint8_t)cmd;
    return 1;
}

/**
 * 获取命令名称
 */
const char* wakeup_command_name(wakeup_command_t cmd);

#endif /* WAKEUP_COMM_H */
```

```c
// wakeup_comm.c
#include "wakeup_comm.h"

const char* wakeup_command_name(wakeup_command_t cmd) {
    switch (cmd) {
        case WAKEUP_CMD_REQUEST_TELEMETRY: return "REQUEST_TELEMETRY";
        case WAKEUP_CMD_SET_CONFIG:        return "SET_CONFIG";
        case WAKEUP_CMD_SYNC_TIME:         return "SYNC_TIME";
        case WAKEUP_CMD_REBOOT:            return "REBOOT";
        default: return "UNKNOWN";
    }
}
```

### 5.2 更换管理员 (ChangeAdmin)

```c
/* 更换管理员 - 需要公司私钥签名 */
bool change_device_admin(serial_port_t *port, uint32_t device_node_id,
                         const uint8_t company_privkey[32],
                         const uint8_t new_gateway_pubkey[32],
                         uint32_t new_gateway_node_id)
{
    /* 1. 获取时间戳 */
    uint64_t timestamp = get_current_timestamp();
    
    /* 2. 构造签名数据 (44字节) */
    uint8_t sign_data[44];
    memcpy(sign_data, new_gateway_pubkey, 32);
    memcpy(sign_data + 32, &timestamp, 8);
    memcpy(sign_data + 40, &device_node_id, 4);
    
    /* 3. Ed25519 签名 */
    uint8_t signature[64];
    if (!ed25519_sign(company_privkey, sign_data, 44, signature)) {
        return false;
    }
    
    /* 4. 构建请求 */
    uint8_t payload[200];
    uint16_t payload_len = private_config_build_change_admin(
        new_gateway_pubkey,
        new_gateway_node_id,
        timestamp,
        signature,
        payload, sizeof(payload));
    
    if (payload_len == 0) {
        return false;
    }
    
    /* 5. 构建 MeshPacket 并发送 */
    packet_build_params_t params;
    packet_build_params_init(&params);
    params.to = device_node_id;
    params.portnum = PORTNUM_PRIVATE_CONFIG_APP;
    params.want_ack = true;
    params.want_response = true;
    params.payload = payload;
    params.payload_len = payload_len;
    
    uint8_t packet[512];
    uint16_t packet_len = mesh_packet_build(&params, packet, sizeof(packet));
    
    return toradio_send_packet(packet, packet_len, serial_write, port);
}
```

## 8. 核心架构说明

### 8.1 网关作为管理员模式

本系统的核心设计是**网关节点即管理员**：

| 方面 | 说明 |
|------|------|
| 管理员身份 | 网关物理上是管理员（admin_key[0] = 网关公钥） |
| Admin命令发送 | 使用网关自己的节点私钥，走原生 PKI 路径 |
| Session Passkey | 由 Meshtastic 原生 AdminModule 自动处理 |
| 入网流程 | SetCompanyKey 命令，无需签名验证 |
| 更换管理员 | 需要公司私钥 Ed25519 签名验证 |

### 8.2 与虚拟身份模式的区别

| 方面 | 虚拟身份模式 (旧) | 网关管理员模式 (新) |
|------|------------------|-------------------|
| from 字段 | 伪造为虚拟节点ID | 使用网关真实节点ID |
| 私钥来源 | 虚拟身份私钥 | 网关节点私钥 |
| PKI 加密 | 手动实现 | Meshtastic 原生处理 |
| Session Passkey | 手动注入 | 原生 AdminModule 处理 |
| admin_key 设置 | 通过 Admin 命令 | SetCompanyKey 直接设置 |
