## 2. 心跳机制设计

### 2.1 需求分析

**为什么需要心跳？**
- USB 串口长时间无数据可能进入省电模式
- 设备需要感知主机是否在线
- 固件 watch dog 约 10 分钟

**心跳间隔选择：**
- 固件 watchdog: ~10 分钟
- 安全间隔: 10-14 分钟 (留有安全余量)
- 推荐值: 10 分钟 (600秒)

### 2.2 心跳包类型选择

| 方案 | 优点 | 缺点 | 推荐度 |
|------|------|------|--------|
| `ToRadio.want_config_id` | 触发配置同步，轻量 | 会收到大量配置数据 | ★★★★★ |
| 空文本消息 | 简单 | 会产生日志，骚扰其他节点 | ★★☆☆☆ |
| 空路由包 (hop_limit=0) | 不产生日志 | 可能被忽略 | ★★★☆☆ |
| `Admin GET_TIME` | 有意义 | 需要权限 | ★★★☆☆ |

**推荐方案：`ToRadio.want_config_id`**

### 2.3 心跳参数

```c
// heartbeat.h
#ifndef HEARTBEAT_H
#define HEARTBEAT_H

#include <stdint.h>
#include <stdbool.h>
#include <time.h>

/* 心跳参数 */
#define HEARTBEAT_INTERVAL_MIN_SEC    600    /* 10分钟 */
#define HEARTBEAT_INTERVAL_MAX_SEC    840    /* 14分钟 */
#define HEARTBEAT_TIMEOUT_SEC         30     /* 响应超时 */
#define HEARTBEAT_MAX_RETRY           3      /* 最大重试次数 */

/* 心跳状态 */
typedef enum {
    HEARTBEAT_STATE_IDLE,
    HEARTBEAT_STATE_WAITING_RESPONSE,
    HEARTBEAT_STATE_RETRYING,
    HEARTBEAT_STATE_FAILED,
} heartbeat_state_t;

/* 心跳管理器 */
typedef struct {
    heartbeat_state_t state;
    uint32_t interval_sec;
    uint32_t last_sent_ms;
    uint32_t last_received_ms;
    uint32_t current_config_id;
    uint8_t retry_count;
    bool enabled;
    
    /* 回调 */
    void (*on_success)(void *user_data);
    void (*on_timeout)(void *user_data);
    void (*on_failed)(void *user_data);
    void *user_data;
} heartbeat_manager_t;

/**
 * 初始化心跳管理器
 */
void heartbeat_init(heartbeat_manager_t *hb);

/**
 * 启动心跳
 */
void heartbeat_start(heartbeat_manager_t *hb);

/**
 * 停止心跳
 */
void heartbeat_stop(heartbeat_manager_t *hb);

/**
 * 心跳定时检查 (在主循环中调用)
 * 
 * @param hb       心跳管理器
 * @param write_fn 写入函数
 * @param user_data 用户数据
 * @return true=发送了心跳, false=未发送
 */
bool heartbeat_tick(heartbeat_manager_t *hb,
                    int (*write_fn)(const uint8_t *data, uint16_t len, void *user_data),
                    void *user_data);

/**
 * 收到数据时调用 (重置心跳计时器)
 */
void heartbeat_on_data_received(heartbeat_manager_t *hb);

/**
 * 收到 config_complete_id 时调用
 */
void heartbeat_on_response(heartbeat_manager_t *hb, uint32_t config_id);

/**
 * 检查心跳是否正常
 */
bool heartbeat_is_healthy(heartbeat_manager_t *hb);

#endif /* HEARTBEAT_H */
```

### 2.4 心跳实现

```c
// heartbeat.c
#include "heartbeat.h"
#include "frame_builder.h"
#include "toradio_builder.h"
#include <stdlib.h>
#include <string.h>

/* 获取当前时间 (毫秒) */
static uint32_t get_current_ms(void) {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (uint32_t)(ts.tv_sec * 1000 + ts.tv_nsec / 1000000);
}

/* 生成随机间隔 */
static uint32_t random_interval(void) {
    uint32_t range = HEARTBEAT_INTERVAL_MAX_SEC - HEARTBEAT_INTERVAL_MIN_SEC;
    return HEARTBEAT_INTERVAL_MIN_SEC + (rand() % range);
}

/* 生成随机 config_id */
static uint32_t random_config_id(void) {
    uint32_t id = (uint32_t)rand();
    return (id == 0) ? 1 : id;
}

void heartbeat_init(heartbeat_manager_t *hb) {
    if (hb == NULL) return;
    
    memset(hb, 0, sizeof(heartbeat_manager_t));
    hb->state = HEARTBEAT_STATE_IDLE;
    hb->interval_sec = random_interval();
    hb->enabled = false;
    hb->retry_count = 0;
}

void heartbeat_start(heartbeat_manager_t *hb) {
    if (hb == NULL) return;
    
    hb->enabled = true;
    hb->state = HEARTBEAT_STATE_IDLE;
    hb->last_sent_ms = get_current_ms();
    hb->last_received_ms = hb->last_sent_ms;
    hb->retry_count = 0;
    hb->interval_sec = random_interval();
}

void heartbeat_stop(heartbeat_manager_t *hb) {
    if (hb == NULL) return;
    hb->enabled = false;
    hb->state = HEARTBEAT_STATE_IDLE;
}

bool heartbeat_tick(heartbeat_manager_t *hb,
                    int (*write_fn)(const uint8_t *data, uint16_t len, void *user_data),
                    void *user_data) {
    if (hb == NULL || !hb->enabled || write_fn == NULL) {
        return false;
    }
    
    uint32_t now = get_current_ms();
    
    switch (hb->state) {
        case HEARTBEAT_STATE_IDLE: {
            /* 检查是否需要发送心跳 */
            uint32_t elapsed_sec = (now - hb->last_received_ms) / 1000;
            
            if (elapsed_sec >= hb->interval_sec) {
                /* 发送心跳 */
                hb->current_config_id = random_config_id();
                
                if (toradio_send_want_config(hb->current_config_id, write_fn, user_data)) {
                    hb->last_sent_ms = now;
                    hb->state = HEARTBEAT_STATE_WAITING_RESPONSE;
                    return true;
                }
            }
            break;
        }
        
        case HEARTBEAT_STATE_WAITING_RESPONSE: {
            /* 检查超时 */
            uint32_t elapsed_sec = (now - hb->last_sent_ms) / 1000;
            
            if (elapsed_sec >= HEARTBEAT_TIMEOUT_SEC) {
                hb->retry_count++;
                
                if (hb->retry_count >= HEARTBEAT_MAX_RETRY) {
                    /* 心跳失败 */
                    hb->state = HEARTBEAT_STATE_FAILED;
                    
                    if (hb->on_failed) {
                        hb->on_failed(hb->user_data);
                    }
                } else {
                    /* 重试 */
                    hb->current_config_id = random_config_id();
                    
                    if (toradio_send_want_config(hb->current_config_id, write_fn, user_data)) {
                        hb->last_sent_ms = now;
                        hb->state = HEARTBEAT_STATE_RETRYING;
                        
                        if (hb->on_timeout) {
                            hb->on_timeout(hb->user_data);
                        }
                    }
                }
            }
            break;
        }
        
        case HEARTBEAT_STATE_RETRYING: {
            /* 检查重试超时 */
            uint32_t elapsed_sec = (now - hb->last_sent_ms) / 1000;
            
            if (elapsed_sec >= HEARTBEAT_TIMEOUT_SEC) {
                hb->retry_count++;
                
                if (hb->retry_count >= HEARTBEAT_MAX_RETRY) {
                    hb->state = HEARTBEAT_STATE_FAILED;
                    
                    if (hb->on_failed) {
                        hb->on_failed(hb->user_data);
                    }
                } else {
                    /* 继续重试 */
                    hb->current_config_id = random_config_id();
                    toradio_send_want_config(hb->current_config_id, write_fn, user_data);
                    hb->last_sent_ms = now;
                }
            }
            break;
        }
        
        case HEARTBEAT_STATE_FAILED:
            /* 等待外部重置 */
            break;
    }
    
    return false;
}

void heartbeat_on_data_received(heartbeat_manager_t *hb) {
    if (hb == NULL) return;
    hb->last_received_ms = get_current_ms();
}

void heartbeat_on_response(heartbeat_manager_t *hb, uint32_t config_id) {
    if (hb == NULL) return;
    
    if (config_id == hb->current_config_id) {
        /* 心跳成功 */
        hb->state = HEARTBEAT_STATE_IDLE;
        hb->retry_count = 0;
        hb->last_received_ms = get_current_ms();
        hb->interval_sec = random_interval();  /* 重新随机间隔 */
        
        if (hb->on_success) {
            hb->on_success(hb->user_data);
        }
    }
}

bool heartbeat_is_healthy(heartbeat_manager_t *hb) {
    if (hb == NULL) return false;
    return hb->state != HEARTBEAT_STATE_FAILED;
}
```

### 3.1 状态定义

```c
#ifndef CONNECTION_MANAGER_H

    CONN_STATE_CONNECTING,
    CONN_STATE_CONNECTED,
    CONN_STATE_HANDSHAKING,
    CONN_STATE_READY,
```

### 3.2 状态机实现

```c
// connection_manager.c
#include "connection_manager.h"
#include <string.h>

static const connection_config_t default_config = {
    .handshake_timeout_ms = 5000,
    .reconnect_delay_ms = 1000,
    .max_reconnect_attempts = 5,
};

const char* connection_state_name(connection_state_t state) {
    static const char* names[] = {
        [CONN_STATE_DISCONNECTED] = "DISCONNECTED",
        [CONN_STATE_CONNECTING] = "CONNECTING",
        [CONN_STATE_CONNECTED] = "CONNECTED",
        [CONN_STATE_HANDSHAKING] = "HANDSHAKING",
        [CONN_STATE_READY] = "READY",
        [CONN_STATE_ERROR] = "ERROR",
        [CONN_STATE_RECONNECTING] = "RECONNECTING",
    };
    
    if (state < sizeof(names) / sizeof(names[0])) {
        return names[state];
    }
    return "UNKNOWN";
}

static void set_state(connection_manager_t *cm, connection_state_t new_state) {
    if (cm->state != new_state) {
        cm->state = new_state;
        
        if (cm->on_state_changed) {
            cm->on_state_changed(new_state, cm->user_data);
        }
    }
}

void connection_manager_init(connection_manager_t *cm, const connection_config_t *config) {
    if (cm == NULL) return;
    
    memset(cm, 0, sizeof(connection_manager_t));
    
    if (config) {
        memcpy(&cm->config, config, sizeof(connection_config_t));
    } else {
        memcpy(&cm->config, &default_config, sizeof(connection_config_t));
    }
    
    heartbeat_init(&cm->heartbeat);
    cm->state = CONN_STATE_DISCONNECTED;
}

void connection_manager_connect(connection_manager_t *cm) {
    if (cm == NULL) return;
    
    cm->reconnect_attempts = 0;
    set_state(cm, CONN_STATE_CONNECTING);
}

void connection_manager_disconnect(connection_manager_t *cm) {
    if (cm == NULL) return;
    
    heartbeat_stop(&cm->heartbeat);
    set_state(cm, CONN_STATE_DISCONNECTED);
}

void connection_manager_handle_event(connection_manager_t *cm, connection_event_t event) {
    if (cm == NULL) return;
    
    switch (cm->state) {
        case CONN_STATE_DISCONNECTED:
            /* 忽略大部分事件 */
            break;
            
        case CONN_STATE_CONNECTING:
            switch (event) {
                case CONN_EVENT_CONNECT:
                    set_state(cm, CONN_STATE_CONNECTED);
                    break;
                case CONN_EVENT_ERROR:
                    set_state(cm, CONN_STATE_ERROR);
                    break;
                default:
                    break;
            }
            break;
            
        case CONN_STATE_CONNECTED:
            switch (event) {
                case CONN_EVENT_DISCONNECT:
                    set_state(cm, CONN_STATE_DISCONNECTED);
                    break;
                case CONN_EVENT_CONNECT:
                    /* 开始握手 */
                    set_state(cm, CONN_STATE_HANDSHAKING);
                    break;
                default:
                    break;
            }
            break;
            
        case CONN_STATE_HANDSHAKING:
            switch (event) {
                case CONN_EVENT_HANDSHAKE_COMPLETE:
                    /* 握手完成，启动心跳 */
                    heartbeat_start(&cm->heartbeat);
                    set_state(cm, CONN_STATE_READY);
                    break;
                case CONN_EVENT_HANDSHAKE_TIMEOUT:
                case CONN_EVENT_DISCONNECT:
                    set_state(cm, CONN_STATE_RECONNECTING);
                    break;
                default:
                    break;
            }
            break;
            
        case CONN_STATE_READY:
            switch (event) {
                case CONN_EVENT_DISCONNECT:
                    heartbeat_stop(&cm->heartbeat);
                    set_state(cm, CONN_STATE_RECONNECTING);
                    break;
                case CONN_EVENT_HEARTBEAT_FAILED:
                    heartbeat_stop(&cm->heartbeat);
                    set_state(cm, CONN_STATE_RECONNECTING);
                    break;
                default:
                    break;
            }
            break;
            
        case CONN_STATE_ERROR:
            /* 等待外部重置 */
            break;
            
        case CONN_STATE_RECONNECTING:
            switch (event) {
                case CONN_EVENT_CONNECT:
                    cm->reconnect_attempts = 0;
                    set_state(cm, CONN_STATE_CONNECTED);
                    break;
                case CONN_EVENT_ERROR:
                    cm->reconnect_attempts++;
                    if (cm->reconnect_attempts >= cm->config.max_reconnect_attempts) {
                        set_state(cm, CONN_STATE_ERROR);
                    }
                    break;
                case CONN_EVENT_DISCONNECT:
                    set_state(cm, CONN_STATE_DISCONNECTED);
                    break;
                default:
                    break;
            }
            break;
    }
}

bool connection_manager_is_connected(connection_manager_t *cm) {
    if (cm == NULL) return false;
    return cm->state >= CONN_STATE_CONNECTED && cm->state != CONN_STATE_ERROR;
}

bool connection_manager_is_ready(connection_manager_t *cm) {
    if (cm == NULL) return false;
    return cm->state == CONN_STATE_READY;
}
```

## 4. 断线重连机制

### 4.1 重连策略

```c
/* 指数退避重连 */
typedef struct {
    uint32_t initial_delay_ms;
    uint32_t max_delay_ms;
    uint32_t current_delay_ms;
    uint32_t max_attempts;
    uint32_t current_attempt;
} reconnect_strategy_t;

void reconnect_strategy_init(reconnect_strategy_t *s) {
    s->initial_delay_ms = 1000;
    s->max_delay_ms = 30000;
    s->current_delay_ms = s->initial_delay_ms;
    s->max_attempts = 5;
    s->current_attempt = 0;
}

bool reconnect_strategy_should_retry(reconnect_strategy_t *s) {
    if (s->current_attempt >= s->max_attempts) {
        return false;
    }
    s->current_attempt++;
    return true;
}

uint32_t reconnect_strategy_get_delay(reconnect_strategy_t *s) {
    uint32_t delay = s->current_delay_ms;
    
    /* 指数退避 */
    s->current_delay_ms *= 2;
    if (s->current_delay_ms > s->max_delay_ms) {
        s->current_delay_ms = s->max_delay_ms;
    }
    
    return delay;
}

void reconnect_strategy_reset(reconnect_strategy_t *s) {
    s->current_delay_ms = s->initial_delay_ms;
    s->current_attempt = 0;
}
```

## 5. 完整示例

```c
/* main.c - 完整连接管理示例 */

#include "frame_parser.h"
#include "frame_builder.h"
#include "connection_manager.h"
#include "from_radio_parser.h"
#include "serial_port.h"

static connection_manager_t g_conn;
static frame_parser_t g_parser;
static uint8_t g_frame_buffer[2048];
static uint8_t g_text_buffer[4096];
static serial_port_t g_serial;

/* 状态变化回调 */
static void on_state_changed(connection_state_t state, void *user_data) {
    printf("Connection state: %s\n", connection_state_name(state));
    
    if (state == CONN_STATE_READY) {
        printf("Ready to communicate!\n");
    } else if (state == CONN_STATE_ERROR) {
        printf("Connection error, please reconnect manually.\n");
    }
}

/* 心跳失败回调 */
static void on_heartbeat_failed(void *user_data) {
    printf("Heartbeat failed, reconnecting...\n");
    connection_manager_handle_event(&g_conn, CONN_EVENT_HEARTBEAT_FAILED);
}

/* 处理帧 */
static void handle_frame(const uint8_t *payload, uint16_t len) {
    from_radio_t fr;
    
    if (from_radio_parse(payload, len, &fr)) {
        /* 通知心跳管理器收到数据 */
        heartbeat_on_data_received(&g_conn.heartbeat);
        
        switch (fr.which_variant) {
            case FROM_RADIO_VARIANT_CONFIG_COMPLETE_ID:
                /* 握手完成或心跳响应 */
                if (g_conn.state == CONN_STATE_HANDSHAKING) {
                    connection_manager_handle_event(&g_conn, CONN_EVENT_HANDSHAKE_COMPLETE);
                } else {
                    heartbeat_on_response(&g_conn.heartbeat, fr.variant.config_complete_id);
                }
                break;
                
            case FROM_RADIO_VARIANT_REBOOTED:
                printf("Device rebooted, re-handshaking...\n");
                connection_manager_handle_event(&g_conn, CONN_EVENT_DISCONNECT);
                break;
                
            default:
                /* 处理其他消息 */
                break;
        }
    }
}

/* 主循环 */
void main_loop(void) {
    /* 初始化 */
    connection_config_t config = {
        .handshake_timeout_ms = 5000,
        .reconnect_delay_ms = 1000,
        .max_reconnect_attempts = 5,
    };
    
    connection_manager_init(&g_conn, &config);
    g_conn.on_state_changed = on_state_changed;
    g_conn.heartbeat.on_failed = on_heartbeat_failed;
    
    frame_parser_init(&g_parser, g_frame_buffer, sizeof(g_frame_buffer),
                      g_text_buffer, sizeof(g_text_buffer));
    
    /* 打开串口 */
    if (!serial_open(&g_serial, "COM3", 115200)) {
        printf("Failed to open serial port\n");
        return;
    }
    
    /* 开始连接 */
    connection_manager_connect(&g_conn);
    
    /* 发送唤醒序列 */
    frame_send_wakeup(serial_write, &g_serial);
    delay_ms(100);
    
    /* 发送握手请求 */
    toradio_send_want_config(generate_config_id(), serial_write, &g_serial);
    connection_manager_handle_event(&g_conn, CONN_EVENT_CONNECT);
    
    /* 主循环 */
    while (running) {
        uint8_t byte;
        
        /* 读取数据 */
        if (serial_read_byte(&g_serial, &byte, 100)) {
            uint8_t *payload;
            uint16_t len;
            
            frame_result_t result = frame_parser_feed(&g_parser, byte, &payload, &len);
            
            if (result == FRAME_RESULT_COMPLETE) {
                handle_frame(payload, len);
            }
        }
        
        /* 心跳检查 */
        heartbeat_tick(&g_conn.heartbeat, serial_write, &g_serial);
        
        /* 检查握手超时 */
        if (g_conn.state == CONN_STATE_HANDSHAKING) {
            static uint32_t handshake_start = 0;
            if (handshake_start == 0) {
                handshake_start = get_current_ms();
            } else if (get_current_ms() - handshake_start > g_conn.config.handshake_timeout_ms) {
                connection_manager_handle_event(&g_conn, CONN_EVENT_HANDSHAKE_TIMEOUT);
                handshake_start = 0;
            }
        }
    }
    
    /* 清理 */
    connection_manager_disconnect(&g_conn);
    serial_close(&g_serial);
}
```
