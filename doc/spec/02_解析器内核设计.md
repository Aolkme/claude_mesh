## 2. FromRadio 解析

### 2.1 消息结构

```protobuf
// mesh.proto
message FromRadio {
    fixed32 id = 1;  // 随机ID, 用于调试
    
    oneof payload_variant {
        MeshPacket packet = 2;
        MyNodeInfo my_info = 3;
        NodeInfo node_info = 4;
        Config config = 5;
        LogRecord log_record = 6;
        uint32 config_complete_id = 7;
        bool rebooted = 8;
        Channel channel = 9;
        QueueStatus queue_status = 10;
        ModuleConfig moduleConfig = 11;
        DeviceMetadata metadata = 12;
        ClientNotification clientNotification = 13;
        DeviceConnectionStatus deviceConnectionStatus = 14;
        FileInfo fileInfo = 15;
        uint32 client_history_timestamp = 16;
    }
}
```

### 2.2 Payload Variant 类型

| 字段号 | variant名 | 消息类型 | 说明 |
|--------|-----------|----------|------|
| 2 | packet | MeshPacket | 数据包 (最常用) |
| 3 | my_info | MyNodeInfo | 本机节点信息 |
| 4 | node_info | NodeInfo | 其他节点信息 |
| 5 | config | Config | 设备配置 |
| 6 | log_record | LogRecord | 设备日志 |
| 7 | config_complete_id | uint32 | 配置同步完成标志 |
| 8 | rebooted | bool | 设备重启标志 |
| 9 | channel | Channel | 信道配置 |
| 10 | queue_status | QueueStatus | 发送队列状态 |
| 11 | moduleConfig | ModuleConfig | 模块配置 |
| 12 | metadata | DeviceMetadata | 固件元数据 |
| 13 | clientNotification | ClientNotification | 客户端通知 |
| 14 | deviceConnectionStatus | DeviceConnectionStatus | 设备连接状态 |
| 15 | fileInfo | FileInfo | 文件信息 |

### 2.4 解析实现 (使用 nanopb)

```c
// from_radio.c
#include "from_radio.h"
#include <string.h>

/* nanopb 字段定义在 mesh.pb.h 中 */
#include "mesh.pb.h"

bool from_radio_parse(const uint8_t *data, uint16_t len, from_radio_t *out) {
    if (data == NULL || len == 0 || out == NULL) {
        return false;
    }
    
    /* 初始化 */
    memset(out, 0, sizeof(from_radio_t));
    
    /* 创建 nanopb 输入流 */
    pb_istream_t stream = pb_istream_from_buffer(data, len);
    
    /* 使用 nanopb 生成的结构直接解码 */
    meshtastic_FromRadio fr = meshtastic_FromRadio_init_zero;
    
    if (!pb_decode(&stream, meshtastic_FromRadio_fields, &fr)) {
        return false;
    }
    
    /* 复制 id */
    out->id = fr.id;
    
    /* 根据变体类型复制数据 */
    out->which_variant = (from_radio_variant_t)fr.which_payload_variant;
    
    switch (fr.which_payload_variant) {
        case meshtastic_FromRadio_packet_tag:
            memcpy(&out->variant.packet, &fr.packet, sizeof(meshtastic_MeshPacket));
            break;
            
        case meshtastic_FromRadio_my_info_tag:
            memcpy(&out->variant.my_info, &fr.my_info, sizeof(meshtastic_MyNodeInfo));
            break;
            
        case meshtastic_FromRadio_node_info_tag:
            memcpy(&out->variant.node_info, &fr.node_info, sizeof(meshtastic_NodeInfo));
            break;
            
        case meshtastic_FromRadio_config_tag:
            memcpy(&out->variant.config, &fr.config, sizeof(meshtastic_Config));
            break;
            
        case meshtastic_FromRadio_log_record_tag:
            memcpy(&out->variant.log_record, &fr.log_record, sizeof(meshtastic_LogRecord));
            break;
            
        case meshtastic_FromRadio_config_complete_id_tag:
            out->variant.config_complete_id = fr.config_complete_id;
            break;
            
        case meshtastic_FromRadio_rebooted_tag:
            out->variant.rebooted = fr.rebooted;
            break;
            
        case meshtastic_FromRadio_channel_tag:
            memcpy(&out->variant.channel, &fr.channel, sizeof(meshtastic_Channel));
            break;
            
        case meshtastic_FromRadio_queue_status_tag:
            memcpy(&out->variant.queue_status, &fr.queue_status, sizeof(meshtastic_QueueStatus));
            break;
            
        case meshtastic_FromRadio_moduleConfig_tag:
            memcpy(&out->variant.module_config, &fr.moduleConfig, sizeof(meshtastic_ModuleConfig));
            break;
            
        case meshtastic_FromRadio_metadata_tag:
            memcpy(&out->variant.metadata, &fr.metadata, sizeof(meshtastic_DeviceMetadata));
            break;
            
        default:
            out->which_variant = FROM_RADIO_VARIANT_NONE;
            break;
    }
    
    return true;
}

const char* from_radio_variant_name(from_radio_variant_t variant) {
    static const char* names[] = {
        [FROM_RADIO_VARIANT_NONE] = "none",
        [FROM_RADIO_VARIANT_PACKET] = "packet",
        [FROM_RADIO_VARIANT_MY_INFO] = "my_info",
        [FROM_RADIO_VARIANT_NODE_INFO] = "node_info",
        [FROM_RADIO_VARIANT_CONFIG] = "config",
        [FROM_RADIO_VARIANT_LOG_RECORD] = "log_record",
        [FROM_RADIO_VARIANT_CONFIG_COMPLETE_ID] = "config_complete_id",
        [FROM_RADIO_VARIANT_REBOOTED] = "rebooted",
        [FROM_RADIO_VARIANT_CHANNEL] = "channel",
        [FROM_RADIO_VARIANT_QUEUE_STATUS] = "queue_status",
        [FROM_RADIO_VARIANT_MODULE_CONFIG] = "moduleConfig",
        [FROM_RADIO_VARIANT_METADATA] = "metadata",
        [FROM_RADIO_VARIANT_CLIENT_NOTIFICATION] = "clientNotification",
        [FROM_RADIO_VARIANT_DEVICE_CONNECTION_STATUS] = "deviceConnectionStatus",
        [FROM_RADIO_VARIANT_FILE_INFO] = "fileInfo",
    };
    
    if (variant < sizeof(names) / sizeof(names[0]) && names[variant]) {
        return names[variant];
    }
    return "unknown";
}
```

## 3. MeshPacket 解析

### 3.1 消息结构

```protobuf
// mesh.proto
message MeshPacket {
    uint32 from = 1;          // 发送方节点号
    uint32 to = 2;            // 接收方节点号 (0xFFFFFFFF = 广播)
    uint32 channel = 3;       // 信道索引
    uint32 id = 4;            // 包ID
    uint32 hop_limit = 5;     // 剩余跳数
    uint32 hop_start = 6;     // 发送时原始跳数
    bool want_ack = 7;        // 需要ACK
    uint32 priority = 8;      // 优先级
    int32 rx_time = 9;        // 接收时间 (Unix时间戳)
    float rx_snr = 10;        // 信噪比
    int32 rx_rssi = 11;       // 信号强度
    int32 delayed = 12;       // 延迟发送秒数
    
    oneof payload_variant {
        Data decoded = 13;    // 明文数据
        bytes encrypted = 14; // 密文数据
    }
}

message Data {
    uint32 portnum = 1;       // 端口号
    bytes payload = 2;        // payload数据
    bool want_response = 3;   // 需要响应
    uint32 reply_id = 4;      // 回复的消息ID
    uint32 emoji = 5;         // Emoji码点
}
```

### 3.2 字段映射规则

| Proto字段 | JSON字段 | 转换规则 | 说明 |
|-----------|----------|----------|------|
| from | from_id | `sprintf(buf, "!%08x", from)` | 转换为节点ID字符串 |
| to | to_id | `0xFFFFFFFF → "broadcast"`<br>`其他 → "!%08x"` | 广播地址特殊处理 |
| hop_start | hop_start | 直接复制 | |
| hop_limit | hop_limit | 直接复制 | |
| - | hops_traveled | `hop_start - hop_limit` | 计算已过跳数 |
| decoded | decoded | 解析后对象 | |
| encrypted | encrypted_hex | `bytes_to_hex()` | 转十六进制字符串 |

### 3.3 C 结构定义

```c
// mesh_packet.h
#ifndef MESH_PACKET_H
#define MESH_PACKET_H

#include <stdint.h>
#include <stdbool.h>

/* 广播地址 */
#define BROADCAST_ADDR      0xFFFFFFFF
#define BROADCAST_NODE_ID   "broadcast"

/* 节点ID转换 */
#define NODE_NUM_TO_ID(buf, num) \
    snprintf((buf), 16, "!%08x", (uint32_t)(num))

/* PortNum 定义 */
typedef enum {
    PORTNUM_UNKNOWN_APP = 0,
    PORTNUM_TEXT_MESSAGE_APP = 1,
    PORTNUM_REMOTE_HARDWARE_APP = 2,
    PORTNUM_POSITION_APP = 3,
    PORTNUM_NODEINFO_APP = 4,
    PORTNUM_ROUTING_APP = 5,
    PORTNUM_ADMIN_APP = 6,
    PORTNUM_TEXT_MESSAGE_COMPRESSED_APP = 7,
    PORTNUM_WAYPOINT_APP = 8,
    PORTNUM_AUDIO_APP = 9,
    PORTNUM_SENSOR_APP = 10,
    PORTNUM_POWERSTRESS_APP = 11,
    PORTNUM_TELEMETRY_APP = 67,
    PORTNUM_TRACEROUTE_APP = 70,
    PORTNUM_NEIGHBORINFO_APP = 71,
    /* 自定义端口 */
    PORTNUM_PRIVATE_APP = 256,
    PORTNUM_CUSTOM_POSITION_APP = 286,
    PORTNUM_PRIVATE_CONFIG_APP = 287,
    PORTNUM_WAKEUP_COMM_APP = 288,
} portnum_t;

/* Data 结构 */
typedef struct {
    portnum_t portnum;
    uint8_t *payload;
    uint16_t payload_len;
    bool want_response;
    uint32_t reply_id;
    uint32_t emoji;
} data_t;

/* MeshPacket 结构 */
typedef struct {
    uint32_t from;
    uint32_t to;
    uint32_t channel;
    uint32_t id;
    uint8_t hop_limit;
    uint8_t hop_start;
    bool want_ack;
    uint8_t priority;
    int32_t rx_time;
    float rx_snr;
    int32_t rx_rssi;
    int32_t delayed;
    
    /* payload variant */
    bool has_decoded;
    bool has_encrypted;
    
    data_t decoded;
    uint8_t *encrypted;
    uint16_t encrypted_len;
    
    /* 扩展字段 (解析后计算) */
    char from_id[16];
    char to_id[16];
    uint8_t hops_traveled;
} mesh_packet_t;

/**
 * 解析 MeshPacket
 */

 * 获取 PortNum 名称
 */

/**
 * 节点号转ID字符串

#endif /* MESH_PACKET_H */
```

### 3.4 解析实现

```c
// mesh_packet.c
#include "mesh_packet.h"
#include "mesh.pb.h"
#include <string.h>

void node_num_to_id(uint32_t num, char *buf, uint16_t buf_size) {
    if (buf == NULL || buf_size < 16) return;
    
    if (num == BROADCAST_ADDR) {
        strncpy(buf, BROADCAST_NODE_ID, buf_size - 1);
        buf[buf_size - 1] = '\0';
    } else {
        snprintf(buf, buf_size, "!%08x", num);
    }
}

const char* portnum_name(portnum_t portnum) {
    static const struct {
        portnum_t num;
        const char *name;
    } portnum_names[] = {
        {PORTNUM_UNKNOWN_APP, "UNKNOWN_APP"},
        {PORTNUM_TEXT_MESSAGE_APP, "TEXT_MESSAGE_APP"},
        {PORTNUM_REMOTE_HARDWARE_APP, "REMOTE_HARDWARE_APP"},
        {PORTNUM_POSITION_APP, "POSITION_APP"},
        {PORTNUM_NODEINFO_APP, "NODEINFO_APP"},
        {PORTNUM_ROUTING_APP, "ROUTING_APP"},
        {PORTNUM_ADMIN_APP, "ADMIN_APP"},
        {PORTNUM_TEXT_MESSAGE_COMPRESSED_APP, "TEXT_MESSAGE_COMPRESSED_APP"},
        {PORTNUM_WAYPOINT_APP, "WAYPOINT_APP"},
        {PORTNUM_AUDIO_APP, "AUDIO_APP"},
        {PORTNUM_SENSOR_APP, "SENSOR_APP"},
        {PORTNUM_POWERSTRESS_APP, "POWERSTRESS_APP"},
        {PORTNUM_TELEMETRY_APP, "TELEMETRY_APP"},
        {PORTNUM_TRACEROUTE_APP, "TRACEROUTE_APP"},
        {PORTNUM_NEIGHBORINFO_APP, "NEIGHBORINFO_APP"},
        {PORTNUM_PRIVATE_APP, "PRIVATE_APP"},
        {PORTNUM_CUSTOM_POSITION_APP, "CUSTOM_POSITION_APP"},
        {PORTNUM_PRIVATE_CONFIG_APP, "PRIVATE_CONFIG_APP"},
        {PORTNUM_WAKEUP_COMM_APP, "WAKEUP_COMM_APP"},
    };
    
    for (int i = 0; i < sizeof(portnum_names) / sizeof(portnum_names[0]); i++) {
        if (portnum_names[i].num == portnum) {
            return portnum_names[i].name;
        }
    }
    
    static char unknown_buf[32];
    snprintf(unknown_buf, sizeof(unknown_buf), "UNKNOWN(%u)", portnum);
    return unknown_buf;
}

bool mesh_packet_parse(const uint8_t *data, uint16_t len, mesh_packet_t *out) {
    if (data == NULL || len == 0 || out == NULL) {
        return false;
    }
    
    memset(out, 0, sizeof(mesh_packet_t));
    
    pb_istream_t stream = pb_istream_from_buffer(data, len);
    meshtastic_MeshPacket mp = meshtastic_MeshPacket_init_zero;
    
    /* 需要设置回调来处理 payload bytes */
    /* 或者使用静态缓冲区 */
    
    if (!pb_decode(&stream, meshtastic_MeshPacket_fields, &mp)) {
        return false;
    }
    
    /* 复制基本字段 */
    out->from = mp.from;
    out->to = mp.to;
    out->channel = mp.channel;
    out->id = mp.id;
    out->hop_limit = mp.hop_limit;
    out->hop_start = mp.hop_start;
    out->want_ack = mp.want_ack;
    out->priority = mp.priority;
    out->rx_time = mp.rx_time;
    out->rx_snr = mp.rx_snr;
    out->rx_rssi = mp.rx_rssi;
    out->delayed = mp.delayed;
    
    /* 计算扩展字段 */
    node_num_to_id(out->from, out->from_id, sizeof(out->from_id));
    node_num_to_id(out->to, out->to_id, sizeof(out->to_id));
    out->hops_traveled = (out->hop_start > out->hop_limit) ? 
                         (out->hop_start - out->hop_limit) : 0;
    
    /* 处理 payload variant */
    if (mp.which_payload_variant == meshtastic_MeshPacket_decoded_tag) {
        out->has_decoded = true;
        out->decoded.portnum = (portnum_t)mp.decoded.portnum;
        out->decoded.want_response = mp.decoded.want_response;
        out->decoded.reply_id = mp.decoded.reply_id;
        out->decoded.emoji = mp.decoded.emoji;
        
        /* payload 需要单独处理 (bytes类型) */
        /* nanopb 需要设置回调或使用固定长度数组 */
    } else if (mp.which_payload_variant == meshtastic_MeshPacket_encrypted_tag) {
        out->has_encrypted = true;
        /* encrypted bytes 处理 */
    }
    
    return true;
}
```

### 4.3 解析实现

```c
// portnum_handler.c
#include "portnum_handler.h"
#include "position.pb.h"
#include "telemetry.pb.h"
#include "admin.pb.h"
#include "routing.pb.h"
#include <stdlib.h>
#include <string.h>

void payload_parse(portnum_t portnum, 
                   const uint8_t *payload, uint16_t len,
                   payload_parse_result_t *result) {
    if (result == NULL) return;
    
    memset(result, 0, sizeof(payload_parse_result_t));
    
    if (payload == NULL || len == 0) {
        result->success = true;
        result->json = strdup("{}");
        return;
    }
    
    /* 文本类端口: UTF-8 解码 */
    if (is_text_portnum(portnum)) {
        /* 分配 JSON 缓冲区: {"text":"..."} */
        uint16_t json_size = len * 2 + 20;
        result->json = malloc(json_size);
        
        if (result->json) {
            /* 转义特殊字符 */
            int pos = snprintf(result->json, json_size, "{\"text\":\"");
            for (uint16_t i = 0; i < len && pos < json_size - 10; i++) {
                uint8_t c = payload[i];
                if (c == '"') {
                    pos += snprintf(result->json + pos, json_size - pos, "\\\"");
                } else if (c == '\\') {
                    pos += snprintf(result->json + pos, json_size - pos, "\\\\");
                } else if (c == '\n') {
                    pos += snprintf(result->json + pos, json_size - pos, "\\n");
                } else if (c == '\r') {
                    pos += snprintf(result->json + pos, json_size - pos, "\\r");
                } else if (c == '\t') {
                    pos += snprintf(result->json + pos, json_size - pos, "\\t");
                } else if (c >= 0x20 && c < 0x7F) {
                    result->json[pos++] = c;
                } else {
                    pos += snprintf(result->json + pos, json_size - pos, "\\x%02x", c);
                }
            }
            snprintf(result->json + pos, json_size - pos, "\"}");
            result->json_len = strlen(result->json);
            result->success = true;
        }
        return;
    }
    
    /* 自定义端口 288: WAKEUP_COMM (原始字节，非proto) */
    if (portnum == PORTNUM_WAKEUP_COMM_APP) {
        /* 1字节命令 */
        result->json = malloc(64);
        if (result->json) {
            snprintf(result->json, 64, "{\"command\":%u,\"command_hex\":\"%02x\"}", 
                     payload[0], payload[0]);
            result->json_len = strlen(result->json);
            result->success = true;
        }
        return;
    }
    
    /* Protobuf 端口 */
    pb_istream_t stream = pb_istream_from_buffer(payload, len);
    
    switch (portnum) {
        case PORTNUM_POSITION_APP:
        case PORTNUM_CUSTOM_POSITION_APP: {
            meshtastic_Position pos = meshtastic_Position_init_zero;
            if (pb_decode(&stream, meshtastic_Position_fields, &pos)) {
                result->json = malloc(512);
                if (result->json) {
                    snprintf(result->json, 512,
                        "{\"latitude_i\":%d,\"longitude_i\":%d,\"altitude\":%d,"
                        "\"time\":%u,\"timestamp\":%u,"
                        "\"location_source\":%u,\"altitude_source\":%u}",
                        pos.latitude_i, pos.longitude_i, pos.altitude,
                        pos.time, pos.timestamp,
                        pos.location_source, pos.altitude_source);
                    result->json_len = strlen(result->json);
                    result->success = true;
                }
            } else {
                snprintf(result->parse_error, sizeof(result->parse_error),
                         "Position decode failed");
            }
            break;
        }
        
        case PORTNUM_NODEINFO_APP: {
            meshtastic_User user = meshtastic_User_init_zero;
            if (pb_decode(&stream, meshtastic_User_fields, &user)) {
                result->json = malloc(512);
                if (result->json) {
                    snprintf(result->json, 512,
                        "{\"id\":\"%s\",\"long_name\":\"%s\",\"short_name\":\"%s\","
                        "\"hw_model\":%u,\"role\":%u}",
                        user.id, user.long_name, user.short_name,
                        user.hw_model, user.role);
                    result->json_len = strlen(result->json);
                    result->success = true;
                }
            }
            break;
        }
        
        case PORTNUM_ROUTING_APP: {
            meshtastic_Routing routing = meshtastic_Routing_init_zero;
            if (pb_decode(&stream, meshtastic_Routing_fields, &routing)) {
                result->json = malloc(128);
                if (result->json) {
                    if (routing.which_variant == meshtastic_Routing_error_reason_tag) {
                        snprintf(result->json, 128, "{\"error_reason\":%d}", 
                                 routing.error_reason);
                    } else {
                        snprintf(result->json, 128, "{\"route_request\":{...}}");
                    }
                    result->json_len = strlen(result->json);
                    result->success = true;
                }
            }
            break;
        }
        
        case PORTNUM_TELEMETRY_APP: {
            meshtastic_Telemetry tel = meshtastic_Telemetry_init_zero;
            if (pb_decode(&stream, meshtastic_Telemetry_fields, &tel)) {
                result->json = malloc(512);
                if (result->json) {
                    int pos = snprintf(result->json, 512, "{\"time\":%u", tel.time);
                    
                    if (tel.which_variant == meshtastic_Telemetry_device_metrics_tag) {
                        pos += snprintf(result->json + pos, 512 - pos,
                            ",\"device_metrics\":{\"battery_level\":%u,\"voltage\":%.2f,"
                            "\"channel_utilization\":%.2f,\"air_util_tx\":%.2f,"
                            "\"uptime_seconds\":%u}}",
                            tel.device_metrics.battery_level, tel.device_metrics.voltage,
                            tel.device_metrics.channel_utilization, tel.device_metrics.air_util_tx,
                            tel.device_metrics.uptime_seconds);
                    } else if (tel.which_variant == meshtastic_Telemetry_environment_metrics_tag) {
                        pos += snprintf(result->json + pos, 512 - pos,
                            ",\"environment_metrics\":{\"temperature\":%.2f,"
                            "\"relative_humidity\":%.2f,\"barometric_pressure\":%.2f}}",
                            tel.environment_metrics.temperature,
                            tel.environment_metrics.relative_humidity,
                            tel.environment_metrics.barometric_pressure);
                    }
                    
                    result->json_len = strlen(result->json);
                    result->success = true;
                }
            }
            break;
        }
        
        case PORTNUM_ADMIN_APP: {
            meshtastic_AdminMessage admin = meshtastic_AdminMessage_init_zero;
            if (pb_decode(&stream, meshtastic_AdminMessage_fields, &admin)) {
                result->json = malloc(256);
                if (result->json) {
                    /* 根据 variant 类型输出 */
                    switch (admin.which_variant) {
                        case meshtastic_AdminMessage_get_config_request_tag:
                            snprintf(result->json, 256, 
                                "{\"get_config_request\":%d}", 
                                admin.get_config_request);
                            break;
                        case meshtastic_AdminMessage_reboot_seconds_tag:
                            snprintf(result->json, 256,
                                "{\"reboot_seconds\":%d}", admin.reboot_seconds);
                            break;
                        case meshtastic_AdminMessage_session_passkey_tag:
                            snprintf(result->json, 256,
                                "{\"session_passkey_hex\":\"%02x%02x%02x%02x%02x%02x%02x%02x\"}",
                                admin.session_passkey[0], admin.session_passkey[1],
                                admin.session_passkey[2], admin.session_passkey[3],
                                admin.session_passkey[4], admin.session_passkey[5],
                                admin.session_passkey[6], admin.session_passkey[7]);
                            break;
                        default:
                            snprintf(result->json, 256, "{\"variant\":%d}", 
                                     admin.which_variant);
                            break;
                    }
                    result->json_len = strlen(result->json);
                    result->success = true;
                }
            }
            break;
        }
        
        case PORTNUM_PRIVATE_CONFIG_APP: {
            /* 自定义协议，见 04_扩展协议规范.md */
            /* 解析 PrivateConfigPacket */
            
            /* 简化处理：输出原始 hex */
            uint16_t json_size = len * 2 + 256;
            result->json = malloc(json_size);
            
            if (result->json) {
                int pos = snprintf(result->json, json_size, 
                    "{\"portnum\":\"PRIVATE_CONFIG_APP\",\"payload_hex\":\"");
                for (uint16_t i = 0; i < len && pos < json_size - 10; i++) {
                    pos += snprintf(result->json + pos, json_size - pos, 
                                    "%02x", payload[i]);
                }
                pos += snprintf(result->json + pos, json_size - pos, "\"");
                
                /* 尝试解析关键字段 */
                /* 检查是否为 response (field 2) */
                /* 检查是否包含 company_config (field 4 in response) */
                
                snprintf(result->json + pos, json_size - pos, "}");
                result->json_len = strlen(result->json);
                result->success = true;
            }
            break;
        }
        
        default: {
            /* 未知类型，输出原始 hex */
            uint16_t json_size = len * 2 + 64;
            result->json = malloc(json_size);
            if (result->json) {
                int pos = snprintf(result->json, json_size, 
                    "{\"portnum\":%u,\"portnum_name\":\"%s\",\"payload_hex\":\"",
                    portnum, portnum_name(portnum));
                for (uint16_t i = 0; i < len && pos < json_size - 4; i++) {
                    pos += snprintf(result->json + pos, json_size - pos, 
                                    "%02x", payload[i]);
                }
                snprintf(result->json + pos, json_size - pos, "\"}");
                result->json_len = strlen(result->json);
                result->success = true;
            }
            break;
        }
    }
}

void payload_parse_result_free(payload_parse_result_t *result) {
    if (result && result->json) {
        free(result->json);
        result->json = NULL;
    }
}
```

## 7. 完整解析流程

```c
/* 完整的帧解析到JSON输出流程 */

void process_frame(const uint8_t *frame_data, uint16_t frame_len) {
    /* 1. 解析 FromRadio */
    from_radio_t fr;
    if (!from_radio_parse(frame_data, frame_len, &fr)) {
        printf("Failed to parse FromRadio\n");
        return;
    }
    
    /* 2. 根据类型处理 */
    switch (fr.which_variant) {
        case FROM_RADIO_VARIANT_PACKET: {
            mesh_packet_t *mp = &fr.variant.packet;
            
            /* 解析 payload */
            if (mp->has_decoded && mp->decoded.payload_len > 0) {
                payload_parse_result_t result;
                payload_parse(mp->decoded.portnum,
                              mp->decoded.payload,
                              mp->decoded.payload_len,
                              &result);
                
                if (result.success) {
                    printf("PortNum: %s\n", portnum_name(mp->decoded.portnum));
                    printf("Payload JSON: %s\n", result.json);
                } else {
                    printf("Parse error: %s\n", result.parse_error);
                }
                
                payload_parse_result_free(&result);
            }
            break;
        }
        
        case FROM_RADIO_VARIANT_MY_INFO:
            printf("MyNodeInfo: node_num=%u\n", fr.variant.my_info.my_node_num);
            break;
            
        case FROM_RADIO_VARIANT_NODE_INFO:
            printf("NodeInfo: num=%u, long_name=%s\n",
                   fr.variant.node_info.num,
                   fr.variant.node_info.user.long_name);
            break;
            
        case FROM_RADIO_VARIANT_CONFIG_COMPLETE_ID:
            printf("Config complete, id=%u\n", fr.variant.config_complete_id);
            break;
            
        default:
            printf("FromRadio variant: %s\n", from_radio_variant_name(fr.which_variant));
            break;
    }
    
    /* 3. 生成 JSON */
    char json_buf[4096];
    uint32_t json_len = json_build_from_radio(&fr, json_buf, sizeof(json_buf));
    if (json_len > 0) {
        printf("Full JSON: %s\n", json_buf);
    }
}
```

## 8. PRIVATE_CONFIG_APP 解析

### 8.1 概述

PRIVATE_CONFIG_APP (PortNum 287) 用于设备入网、更换管理员和同步唤醒配置。

**关键消息类型：**

| 消息 | 字段号 | 说明 |
|------|--------|------|
| PrivateConfigPacket.request | 1 | 请求消息 |
| PrivateConfigPacket.response | 2 | 响应消息 |
| response.private_config | 1 | 私有配置 |
| response.company_config | 4 | 公司配置（入网状态） |

### 8.2 解析 CompanyConfig

```c
/* 解析 CompanyConfig，判断设备入网状态 */
bool parse_company_config(const uint8_t *payload, uint16_t len,
                           bool *is_enrolled, uint8_t *company_pubkey)
{
    if (payload == NULL || len == 0) return false;
    
    /* 简化解析：查找关键字段 */
    /* CompanyConfig 结构:
     *   field 1: company_public_key (bytes, 32字节)
     *   field 2: is_enrolled (bool)
     *   field 3: last_change_timestamp (uint64)
     */
    
    /* 查找 is_enrolled 字段 (field 2, varint) */
    /* tag = (2 << 3) | 0 = 0x10 */
    for (uint16_t i = 0; i < len - 1; i++) {
        if (payload[i] == 0x10) {
            *is_enrolled = (payload[i + 1] != 0);
            break;
        }
    }
    
    /* 查找 company_public_key 字段 (field 1, bytes) */
    /* tag = (1 << 3) | 2 = 0x0a */
    for (uint16_t i = 0; i < len - 35; i++) {
        if (payload[i] == 0x0a && payload[i + 1] == 0x20) {
            /* 0x20 = 32 字节长度 */
            memcpy(company_pubkey, payload + i + 2, 32);
            break;
        }
    }
    
    return true;
}
```

### 8.3 完整解析示例

```c
/* 处理 PRIVATE_CONFIG_APP 响应 */
void handle_private_config_response(const uint8_t *payload, uint16_t len)
{
    /* 1. 判断是 request 还是 response */
    /* PrivateConfigPacket:
     *   field 1 = request (oneof packet_type)
     *   field 2 = response (oneof packet_type)
     */
    
    bool is_response = false;
    for (uint16_t i = 0; i < len; i++) {
        if (payload[i] == 0x12) {  /* tag = (2 << 3) | 2 = 0x12 */
            is_response = true;
            break;
        }
    }
    
    if (!is_response) {
        printf("PRIVATE_CONFIG: Received request (ignoring)\n");
        return;
    }
    
    /* 2. 解析响应内容 */
    bool is_enrolled = false;
    uint8_t company_pubkey[32] = {0};
    
    if (parse_company_config(payload, len, &is_enrolled, company_pubkey)) {
        if (is_enrolled) {
            printf("Device is ENROLLED\n");
            printf("Company pubkey: ");
            for (int i = 0; i < 32; i++) {
                printf("%02x", company_pubkey[i]);
            }
            printf("\n");
        } else {
            printf("Device is NOT enrolled (empty device)\n");
        }
    }
}
```
