# 03_构造器内核设计

> 本文档详细说明如何构造 Meshtastic 数据包，包括 MeshPacket、ToRadio 和各类型 payload 的构建方法。
>
> **重要说明：** 网关作为管理员模式下，使用网关自己的节点私钥发送命令，走原生 PKI 路径。

## 2. 发送流程概览

### 2.1 整体流程

```
用户输入参数
     │
     ▼
┌─────────────────────┐
│ 构造 MeshPacket     │
│ - to, channel, id   │
│ - hop_limit, etc.   │
│ - decoded payload   │
└──────────┬──────────┘
           │
           ▼
┌─────────────────────┐
│ (可选) 注入 Session │
│ Passkey (Admin)     │
└──────────┬──────────┘
           │
           ▼
┌─────────────────────┐
│ (可选) PKI 加密     │
│ - X25519 密钥交换   │
│ - AES-256-CCM       │
└──────────┬──────────┘
           │
           ▼
┌─────────────────────┐
│ 序列化 ToRadio      │
│ - packet 字段       │
└──────────┬──────────┘
           │
           ▼
┌─────────────────────┐
│ 构造帧并发送        │
│ - 0x94 0xC3 header  │
│ - write to serial   │
└─────────────────────┘
```

### 2.3 网关作为管理员模式

**核心架构说明：**

本系统采用"网关即管理员"模式，与传统的"虚拟身份"模式有本质区别：

| 方面 | 虚拟身份模式 (旧) | 网关管理员模式 (本系统) |
|------|------------------|------------------------|
| **管理员身份** | 虚拟身份（伪造 from 字段） | 网关节点物理上是管理员 |
| **admin_key 设置** | 通过 Admin 命令设置 | SetCompanyKey 直接设置 admin_key[0] |
| **Admin 命令发送** | PKI 加密 + Session Passkey 注入 | 使用网关节点私钥，原生 PKI 路径 |
| **Session Passkey** | 手动管理 | Meshtastic 原生 AdminModule 自动处理 |
| **from 字段** | 伪造为虚拟节点 ID | 使用网关真实节点 ID |
| **入网流程** | want_config 握手 | SetCompanyKey 命令 |

**网关管理员模式的优势：**

1. **无需伪造身份** - 网关使用自己的节点密钥
2. **原生 PKI 支持** - Meshtastic 自动处理加密
3. **简化 Session 管理** - AdminModule 自动处理
4. **更安全的入网流程** - 通过公司密钥验证机制

## 3. MeshPacket 构造

### 3.1 默认值定义

```c
// packet_builder.h
#ifndef PACKET_BUILDER_H
#define PACKET_BUILDER_H

#include <stdint.h>
#include <stdbool.h>

/* 默认值 */
#define DEFAULT_HOP_LIMIT      3
#define DEFAULT_HOP_START      3
#define DEFAULT_CHANNEL        0
#define DEFAULT_WANT_ACK       false
#define DEFAULT_PRIORITY       0

/* 广播地址 */
#define BROADCAST_ADDR         0xFFFFFFFF

/* MeshPacket 构建参数 */
typedef struct {
    uint32_t to;                /* 目标节点号 */
    uint32_t channel;           /* 信道索引 */
    uint32_t id;                /* 包ID (0=自动生成) */
    uint8_t hop_limit;          /* 跳数限制 */
    uint8_t hop_start;          /* 发送时跳数 */
    bool want_ack;              /* 需要ACK */
    uint8_t priority;           /* 优先级 */
    int32_t delayed;            /* 延迟发送秒数 */
    
    /* Decoded payload */
    uint32_t portnum;           /* 端口号 */
    const uint8_t *payload;     /* payload 数据 */
    uint16_t payload_len;       /* payload 长度 */
    bool want_response;         /* 需要响应 */
    uint32_t reply_id;          /* 回复ID */
    uint32_t emoji;             /* Emoji */
} packet_build_params_t;

/**
 * 初始化默认参数
 */
void packet_build_params_init(packet_build_params_t *params);

/**
 * 构建 MeshPacket
 * 
 * @param params   构建参数
 * @param out      输出缓冲区
 * @param out_size 缓冲区大小
 * @return 实际长度，0表示失败
 */
uint16_t mesh_packet_build(const packet_build_params_t *params,
                            uint8_t *out, uint16_t out_size);

/**
 * 生成随机包ID
 */
uint32_t generate_packet_id(void);

#endif /* PACKET_BUILDER_H */
```

### 3.2 构建实现

```c
// packet_builder.c
#include "packet_builder.h"
#include "mesh.pb.h"
#include <string.h>
#include <stdlib.h>
#include <time.h>

void packet_build_params_init(packet_build_params_t *params) {
    if (params == NULL) return;
    
    memset(params, 0, sizeof(packet_build_params_t));
    params->to = BROADCAST_ADDR;
    params->channel = DEFAULT_CHANNEL;
    params->id = 0;  /* 自动生成 */
    params->hop_limit = DEFAULT_HOP_LIMIT;
    params->hop_start = DEFAULT_HOP_START;
    params->want_ack = DEFAULT_WANT_ACK;
    params->priority = DEFAULT_PRIORITY;
    params->delayed = 0;
}

uint32_t generate_packet_id(void) {
    static bool seeded = false;
    if (!seeded) {
        srand((unsigned int)time(NULL));
        seeded = true;
    }
    return (uint32_t)rand() | ((uint32_t)rand() << 16);
}

uint16_t mesh_packet_build(const packet_build_params_t *params,
                            uint8_t *out, uint16_t out_size) {
    if (params == NULL || out == NULL || out_size == 0) {
        return 0;
    }
    
    /* 初始化 MeshPacket */
    meshtastic_MeshPacket mp = meshtastic_MeshPacket_init_zero;
    
    /* 设置基本字段 */
    mp.to = params->to;
    mp.channel = params->channel;
    mp.id = (params->id != 0) ? params->id : generate_packet_id();
    mp.hop_limit = params->hop_limit;
    mp.hop_start = params->hop_start;
    mp.want_ack = params->want_ack;
    mp.priority = params->priority;
    mp.delayed = params->delayed;
    
    /* 设置 decoded payload */
    mp.which_payload_variant = meshtastic_MeshPacket_decoded_tag;
    mp.decoded.portnum = params->portnum;
    mp.decoded.want_response = params->want_response;
    mp.decoded.reply_id = params->reply_id;
    mp.decoded.emoji = params->emoji;
    
    /* 复制 payload */
    if (params->payload != NULL && params->payload_len > 0) {
        /* nanopb 需要使用回调或固定数组 */
        /* 这里假设使用固定数组 */
        if (params->payload_len <= sizeof(mp.decoded.payload.bytes)) {
            memcpy(mp.decoded.payload.bytes, params->payload, params->payload_len);
            mp.decoded.payload.size = params->payload_len;
        } else {
            return 0;  /* payload 过长 */
        }
    }
    
    /* 序列化 */
    pb_ostream_t stream = pb_ostream_from_buffer(out, out_size);
    if (!pb_encode(&stream, meshtastic_MeshPacket_fields, &mp)) {
        return 0;
    }
    
    return (uint16_t)stream.bytes_written;
}
```

## 4. Payload 构建

### 4.1 文本消息 (PortNum=1)

```c
// payload_builder.h
#ifndef PAYLOAD_BUILDER_H
#define PAYLOAD_BUILDER_H

#include <stdint.h>
#include <stdbool.h>

/* Position 结构 */
typedef struct {
    double latitude;            /* 纬度 (度) */
    double longitude;           /* 经度 (度) */
    int32_t altitude;           /* 海拔 (米) */
    uint32_t time;              /* Unix时间戳 */
    uint32_t timestamp;
    float pdop;
    float hdop;
    float vdop;
    uint32_t ground_speed;      /* 地速 (cm/s) */
    uint32_t ground_track;      /* 航向 (0.00001度) */
    uint8_t location_source;
    uint8_t altitude_source;
    uint32_t precision_bits;
} position_params_t;

/* User 结构 */
typedef struct {
    const char *id;             /* "!xxxxxxxx" */
    const char *long_name;      /* 长名称 */
    const char *short_name;     /* 短名称 */
    uint8_t hw_model;           /* 硬件型号 */
    uint8_t role;               /* 角色 */
    const uint8_t *public_key;  /* 公钥 (32字节) */
    bool is_licensed;
} user_params_t;

/* Telemetry 结构 */
typedef struct {
    uint32_t time;
    uint32_t battery_level;     /* 0-100, 101=外接电源 */
    float voltage;
    float channel_utilization;
    float air_util_tx;
    uint32_t uptime_seconds;
} device_metrics_t;

/* 构建文本消息 payload */
uint16_t build_text_payload(const char *text, uint8_t *out, uint16_t out_size);

/* 构建位置消息 payload */
uint16_t build_position_payload(const position_params_t *pos, uint8_t *out, uint16_t out_size);

/* 构建用户信息 payload */
uint16_t build_user_payload(const user_params_t *user, uint8_t *out, uint16_t out_size);

/* 构建设备遥测 payload */
uint16_t build_telemetry_payload(const device_metrics_t *metrics, uint8_t *out, uint16_t out_size);

#endif /* PAYLOAD_BUILDER_H */
```

### 4.2 Payload 构建实现

```c
// payload_builder.c
#include "payload_builder.h"
#include "position.pb.h"
#include "telemetry.pb.h"
#include "mesh.pb.h"
#include <string.h>

uint16_t build_text_payload(const char *text, uint8_t *out, uint16_t out_size) {
    if (text == NULL || out == NULL || out_size == 0) {
        return 0;
    }
    
    uint16_t len = strlen(text);
    if (len == 0 || len > out_size) {
        return 0;
    }
    
    memcpy(out, text, len);
    return len;
}

uint16_t build_position_payload(const position_params_t *pos, uint8_t *out, uint16_t out_size) {
    if (pos == NULL || out == NULL || out_size == 0) {
        return 0;
    }
    
    meshtastic_Position pb = meshtastic_Position_init_zero;
    
    /* 纬度/经度: 度 × 1e7 */
    pb.latitude_i = (int32_t)(pos->latitude * 1e7);
    pb.longitude_i = (int32_t)(pos->longitude * 1e7);
    pb.altitude = pos->altitude;
    pb.time = pos->time;
    pb.timestamp = pos->timestamp;
    
    /* 精度因子 */
    pb.PDOP = (uint32_t)(pos->pdop * 100);
    pb.HDOP = (uint32_t)(pos->hdop * 100);
    pb.VDOP = (uint32_t)(pos->vdop * 100);
    
    /* 运动 */
    pb.ground_speed = pos->ground_speed;
    pb.ground_track = pos->ground_track;
    
    /* 来源 */
    pb.location_source = pos->location_source;
    pb.altitude_source = pos->altitude_source;
    pb.precision_bits = pos->precision_bits;
    
    /* 序列化 */
    pb_ostream_t stream = pb_ostream_from_buffer(out, out_size);
    if (!pb_encode(&stream, meshtastic_Position_fields, &pb)) {
        return 0;
    }
    
    return (uint16_t)stream.bytes_written;
}

uint16_t build_user_payload(const user_params_t *user, uint8_t *out, uint16_t out_size) {
    if (user == NULL || out == NULL || out_size == 0) {
        return 0;
    }
    
    meshtastic_User pb = meshtastic_User_init_zero;
    
    /* 复制字符串字段 */
    if (user->id) {
        strncpy(pb.id, user->id, sizeof(pb.id) - 1);
    }
    if (user->long_name) {
        strncpy(pb.long_name, user->long_name, sizeof(pb.long_name) - 1);
    }
    if (user->short_name) {
        strncpy(pb.short_name, user->short_name, sizeof(pb.short_name) - 1);
    }
    
    pb.hw_model = user->hw_model;
    pb.role = user->role;
    pb.is_licensed = user->is_licensed;
    
    /* 公钥 */
    if (user->public_key) {
        memcpy(pb.public_key.bytes, user->public_key, 32);
        pb.public_key.size = 32;
        pb.has_public_key = true;
    }
    
    /* 序列化 */
    pb_ostream_t stream = pb_ostream_from_buffer(out, out_size);
    if (!pb_encode(&stream, meshtastic_User_fields, &pb)) {
        return 0;
    }
    
    return (uint16_t)stream.bytes_written;
}

uint16_t build_telemetry_payload(const device_metrics_t *metrics, uint8_t *out, uint16_t out_size) {
    if (metrics == NULL || out == NULL || out_size == 0) {
        return 0;
    }
    
    meshtastic_Telemetry pb = meshtastic_Telemetry_init_zero;
    
    pb.time = metrics->time;
    pb.which_variant = meshtastic_Telemetry_device_metrics_tag;
    
    pb.device_metrics.battery_level = metrics->battery_level;
    pb.device_metrics.voltage = metrics->voltage;
    pb.device_metrics.channel_utilization = metrics->channel_utilization;
    pb.device_metrics.air_util_tx = metrics->air_util_tx;
    pb.device_metrics.uptime_seconds = metrics->uptime_seconds;
    
    /* 序列化 */
    pb_ostream_t stream = pb_ostream_from_buffer(out, out_size);
    if (!pb_encode(&stream, meshtastic_Telemetry_fields, &pb)) {
        return 0;
    }
    
    return (uint16_t)stream.bytes_written;
}
```

## 6. Session Passkey 管理

### 6.1 概述

Session Passkey 用于 Admin SET 操作的授权：
- 从 Admin GET 响应中自动提取
- 有效期约 270 秒
- 发送 Admin SET 时需要自动注入

### 6.2 网关模式说明

**重要：** 在"网关作为管理员"模式下，Session Passkey 由 Meshtastic 原生 AdminModule 自动处理：

| 操作 | 传统模式 | 网关管理员模式 |
|------|---------|---------------|
| Session Passkey 获取 | 手动从 GET 响应提取 | AdminModule 自动管理 |
| Session Passkey 注入 | 手动编码到 SET 请求 | AdminModule 自动注入 |
| 有效期检查 | 手动实现 | AdminModule 自动处理 |

**结论：** 网关模式下，开发者无需关心 Session Passkey 的管理细节。

### 6.3 传统模式实现（参考）

以下代码供需要手动管理 Session Passkey 的场景参考：

```c
// session_manager.c
#include "session_manager.h"
#include "admin.pb.h"
#include <string.h>

static session_passkey_t g_session = {0};

void session_manager_init(void) {
    memset(&g_session, 0, sizeof(g_session));
}

bool session_manager_extract(const uint8_t *admin_payload, uint16_t len) {
    if (admin_payload == NULL || len == 0) {
        return false;
    }
    
    pb_istream_t stream = pb_istream_from_buffer(admin_payload, len);
    meshtastic_AdminMessage admin = meshtastic_AdminMessage_init_zero;
    
    if (!pb_decode(&stream, meshtastic_AdminMessage_fields, &admin)) {
        return false;
    }
    
    /* 检查是否包含 session_passkey */
    if (admin.which_variant == meshtastic_AdminMessage_session_passkey_tag) {
        memcpy(g_session.passkey, admin.session_passkey.bytes, SESSION_PASSKEY_LEN);
        g_session.valid = true;
        g_session.timestamp = time(NULL);
        return true;
    }
    
    return false;
}

bool session_manager_get(uint8_t *passkey_out) {
    if (passkey_out == NULL) {
        return false;
    }
    
    if (!session_manager_is_valid()) {
        return false;
    }
    
    memcpy(passkey_out, g_session.passkey, SESSION_PASSKEY_LEN);
    return true;
}

bool session_manager_is_valid(void) {
    if (!g_session.valid) {
        return false;
    }
    
    time_t now = time(NULL);
    if (now - g_session.timestamp > SESSION_TIMEOUT_SECS) {
        g_session.valid = false;
        return false;
    }
    
    return true;
}

void session_manager_clear(void) {
    g_session.valid = false;
    memset(g_session.passkey, 0, SESSION_PASSKEY_LEN);
}

int session_manager_remaining_secs(void) {
    if (!g_session.valid) {
        return 0;
    }
    
    time_t now = time(NULL);
    int elapsed = (int)(now - g_session.timestamp);
    int remaining = SESSION_TIMEOUT_SECS - elapsed;
    
    return (remaining > 0) ? remaining : 0;
}
```

## 7. PKI 加密

### 7.1 加密条件

1. 发送 Admin 消息到远程节点
2. 用户选择使用 PKI 加密
3. 已配置虚拟身份私钥 (32字节)
4. 目标节点公钥已知 (32字节)

### 7.2 网关模式说明

**重要：** 在"网关作为管理员"模式下，PKI 加密由 Meshtastic 原生处理：

| 方面 | 传统模式 | 网关管理员模式 |
|------|---------|---------------|
| **私钥来源** | 虚拟身份私钥 | 网关节点私钥 |
| **from 字段** | 伪造为虚拟节点 ID | 使用网关真实节点 ID |
| **加密方式** | 手动实现 X25519 + AES-CCM | Meshtastic 原生 PKI 路径 |
| **nonce 构造** | 手动编码 | 固件自动处理 |

**结论：** 网关模式下，发送 Admin 命令时使用网关自己的节点私钥，固件自动处理 PKI 加密。

### 7.3 加密流程（传统模式参考）

以下流程供需要手动实现 PKI 加密的场景参考：

```
1. X25519 密钥交换
   shared_raw = X25519(private_key, target_public_key)
   shared_key = SHA256(shared_raw)

2. 构造 Nonce (13字节)
   nonce[0:8]  = packet_id (小端)
   nonce[8:12] = from_node_num (小端)
   nonce[12]   = extra_nonce[0]

3. AES-256-CCM 加密
   ciphertext = AES-256-CCM(shared_key, nonce, plaintext)
   
4. 输出格式
   encrypted = ciphertext + 8字节tag + 4字节extra_nonce
```

### 7.4 实现（传统模式参考）

```c
// pki_crypto.c
#include "pki_crypto.h"
#include <string.h>

/* 需要 crypto 库支持，这里提供接口实现示例 */
#ifdef USE_MBEDTLS
#include <mbedtls/x25519.h>
#include <mbedtls/sha256.h>
#include <mbedtls/ccm.h>

bool pki_generate_keypair(keypair_t *keypair) {
    /* 生成随机私钥 */
    mbedtls_x25519_context ctx;
    mbedtls_x25519_init(&ctx);
    
    /* 生成密钥对 */
    int ret = mbedtls_x25519_genkey(&ctx);
    if (ret != 0) {
        mbedtls_x25519_free(&ctx);
        return false;
    }
    
    /* 复制私钥 */
    memcpy(keypair->private_key, ctx.d, PRIVATE_KEY_LEN);
    
    /* 计算公钥 */
    ret = mbedtls_x25519_make_public(&ctx, keypair->public_key);
    
    mbedtls_x25519_free(&ctx);
    return ret == 0;
}

bool pki_x25519_shared_key(const uint8_t *private_key, const uint8_t *public_key,
                            uint8_t *shared_key_out) {
    mbedtls_x25519_context ctx;
    mbedtls_x25519_init(&ctx);
    
    /* 设置私钥 */
    memcpy(ctx.d, private_key, PRIVATE_KEY_LEN);
    
    /* 计算共享密钥 */
    uint8_t shared_raw[32];
    int ret = mbedtls_x25519_shared(&ctx, shared_raw, public_key);
    
    mbedtls_x25519_free(&ctx);
    
    if (ret != 0) {
        return false;
    }
    
    /* SHA256 哈希得到最终密钥 */
    mbedtls_sha256(shared_raw, 32, shared_key_out, 0);
    
    return true;
}

bool pki_encrypt(const pki_encrypt_params_t *params, uint8_t *out, uint16_t *out_len) {
    if (params == NULL || out == NULL || out_len == NULL) {
        return false;
    }
    
    uint16_t required_len = params->plaintext_len + TAG_LEN + EXTRA_NONCE_LEN;
    if (*out_len < required_len) {
        return false;
    }
    
    /* 1. 计算共享密钥 */
    uint8_t shared_key[SHARED_KEY_LEN];
    if (!pki_x25519_shared_key(params->sender_private_key, 
                                params->receiver_public_key, shared_key)) {
        return false;
    }
    
    /* 2. 生成 extra_nonce */
    uint8_t extra_nonce[EXTRA_NONCE_LEN] = {0x01, 0x02, 0x03, 0x04};  /* 随机生成 */
    
    /* 3. 构造 nonce */
    uint8_t nonce[NONCE_LEN];
    /* packet_id (8字节, 小端) */
    uint32_t pid = params->packet_id;
    memcpy(nonce, &pid, 4);
    memset(nonce + 4, 0, 4);  /* 高4字节为0 */
    
    /* from_node_num (4字节, 小端) */
    uint32_t from = params->from_node_num;
    memcpy(nonce + 8, &from, 4);
    
    /* extra_nonce[0] */
    nonce[12] = extra_nonce[0];
    
    /* 4. AES-256-CCM 加密 */
    mbedtls_ccm_context ccm;
    mbedtls_ccm_init(&ccm);
    
    int ret = mbedtls_ccm_setkey(&ccm, MBEDTLS_CIPHER_ID_AES, 
                                  shared_key, 256);
    if (ret != 0) {
        mbedtls_ccm_free(&ccm);
        return false;
    }
    
    /* 加密并生成 tag */
    ret = mbedtls_ccm_encrypt_and_tag(&ccm, params->plaintext_len,
                                       nonce, NONCE_LEN,
                                       NULL, 0,  /* 无附加数据 */
                                       params->plaintext,
                                       out,  /* ciphertext */
                                       out + params->plaintext_len, TAG_LEN);
    
    mbedtls_ccm_free(&ccm);
    
    if (ret != 0) {
        return false;
    }
    
    /* 5. 追加 extra_nonce */
    memcpy(out + params->plaintext_len + TAG_LEN, extra_nonce, EXTRA_NONCE_LEN);
    
    *out_len = required_len;
    return true;
}

bool pki_decrypt(const pki_decrypt_params_t *params, uint8_t *out, uint16_t *out_len) {
    /* 解密逻辑与加密对称 */
    /* 略... */
    return false;
}

/* 无加密库时的空实现 */

bool pki_x25519_shared_key(const uint8_t *private_key, const uint8_t *public_key,
                            uint8_t *shared_key_out) { return false; }
bool pki_encrypt(const pki_encrypt_params_t *params, uint8_t *out, uint16_t *out_len) { 
    return false;
```

```c
    len += 4;
```

```c
    admin_inject_session_passkey(payload, &payload_len, passkey, 8);

    packet_build_params_t params;
    packet_build_params_init(&params);
    params.to = to_node;
    params.portnum = PORTNUM_ADMIN_APP;
    params.want_ack = true;
    params.payload = payload;
    params.payload_len = payload_len;
    
    uint8_t packet[512];
    uint16_t packet_len = mesh_packet_build(&params, packet, sizeof(packet));
    
    return toradio_send_packet(packet, packet_len, serial_write, port);
}
```
