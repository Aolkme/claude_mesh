## 2. 帧格式定义

### 2.1 基本帧结构

```
┌──────────┬──────────┬───────────────┬──────────────────────────────┐
│ START1   │ START2   │ Length (2B)   │ Payload (N bytes)            │
│ 0x94     │ 0xC3     │ Big Endian    │ FromRadio / ToRadio protobuf │
├──────────┼──────────┼───────────────┼──────────────────────────────┤
│ 偏移 0   │ 偏移 1   │ 偏移 2-3      │ 偏移 4+                      │
└──────────┴──────────┴───────────────┴──────────────────────────────┘
```

**示例:**

```
发送空 ToRadio.want_config_id:
帧数据: 94 C3 02 00 08 5A A3 9D 05
        │  │  │  │  └─────────────┘
        │  │  │  │     Payload (ToRadio序列化)
        │  │  │  └─ Length低字节 = 2
        │  │  └─ Length高字节 = 0
        │  └─ START2 = 0xC3
        └─ START1 = 0x94

Payload解析:
08 5A A3 9D 05
│  └──────────┘
│     want_config_id = 12345678 (varint编码)
└─ Tag: field=1, wire_type=0 (varint)
```

### 2.2 常量定义

```c
// frame_protocol.h
#ifndef FRAME_PROTOCOL_H
#define FRAME_PROTOCOL_H

#include <stdint.h>

/* 帧魔数 */
#define FRAME_START1        0x94
#define FRAME_START2        0xC3

/* 帧头长度 */
#define FRAME_HEADER_LEN    4

/* 最大payload长度 */
#define MAX_PAYLOAD_LEN     (512 * 1024)    /* 512KB 防护上限 */
#define MAX_TO_FROM_RADIO   512             /* 标准最大帧长度 (Constants.DATA_PAYLOAD_LEN = 233 + overhead) */

/* 默认波特率 */
#define DEFAULT_BAUDRATE    115200

/* 握手超时 */
#define HANDSHAKE_TIMEOUT_MS    5000

/* 读取超时 */
#define READ_TIMEOUT_MS         500

#endif /* FRAME_PROTOCOL_H */
```

### 2.3 字节序说明

| 字段 | 字节序 | 说明 |
|------|--------|------|
| Length | Big Endian (网络序) | 高字节在前，如长度 512 = 0x0200 → `02 00` |
| Payload | - | Protobuf内部使用小端varint编码 |

## 3. 帧解析状态机

### 3.1 状态定义

```c
// frame_parser.h
#ifndef FRAME_PARSER_H
#define FRAME_PARSER_H

#include <stdint.h>
#include <stdbool.h>

/* 解析器状态 */
typedef enum {
    FRAME_STATE_WAIT_START1,   /* 等待 0x94 */
    FRAME_STATE_WAIT_START2,   /* 等待 0xC3 */
    FRAME_STATE_WAIT_LEN1,     /* 等待长度高字节 */
    FRAME_STATE_WAIT_LEN2,     /* 等待长度低字节 */
    FRAME_STATE_WAIT_PAYLOAD   /* 等待payload数据 */
} frame_parser_state_t;

/* 解析结果 */
typedef enum {
    FRAME_RESULT_NEED_MORE = 0,   /* 需要更多数据 */
    FRAME_RESULT_COMPLETE = 1,    /* 帧完成 */
    FRAME_RESULT_ERROR = -1,      /* 解析错误 */
} frame_result_t;

/* 帧解析器结构体 */
typedef struct {
    frame_parser_state_t state;       /* 当前状态 */
    uint8_t  *buffer;                 /* 接收缓冲区 */
    uint16_t buffer_size;             /* 缓冲区大小 */
    uint16_t payload_len;             /* 期望的payload长度 */
    uint16_t received_len;            /* 已接收长度 */
    
    /* 非帧数据缓冲区 (用于设备输出的文本日志) */
    uint8_t  *text_buffer;
    uint16_t text_buffer_size;
    uint16_t text_len;
} frame_parser_t;

/**
 * 初始化解析器
 * 
 * @param parser         解析器结构体
 * @param buffer         payload接收缓冲区
 * @param buffer_size    缓冲区大小 (建议 >= 2048)
 * @param text_buffer    文本数据缓冲区
 * @param text_buffer_size 文本缓冲区大小 (建议 >= 4096)
 */
void frame_parser_init(frame_parser_t *parser, 
                       uint8_t *buffer, uint16_t buffer_size,
                       uint8_t *text_buffer, uint16_t text_buffer_size);

/**
 * 输入一个字节
 * 
 * @param parser      解析器
 * @param byte        输入字节
 * @param payload_out 输出: payload指针 (仅当返回FRAME_RESULT_COMPLETE时有效)
 * @param len_out     输出: payload长度
 * @return 解析结果
 */
frame_result_t frame_parser_feed(frame_parser_t *parser, uint8_t byte,
                                  uint8_t **payload_out, uint16_t *len_out);

/**
 * 获取缓冲的文本数据
 * 
 * @param parser    解析器
 * @param len_out   输出: 文本数据长度
 * @return 文本数据指针
 */
const uint8_t* frame_parser_get_text(frame_parser_t *parser, uint16_t *len_out);

/**
 * 清空文本缓冲区
 */
void frame_parser_clear_text(frame_parser_t *parser);

/**
 * 重置解析器状态
 */
void frame_parser_reset(frame_parser_t *parser);

#endif /* FRAME_PARSER_H */
```

### 3.2 状态机实现

```c
// frame_parser.c
#include "frame_parser.h"
#include "frame_protocol.h"
#include <string.h>

void frame_parser_init(frame_parser_t *parser, 
                       uint8_t *buffer, uint16_t buffer_size,
                       uint8_t *text_buffer, uint16_t text_buffer_size) {
    parser->state = FRAME_STATE_WAIT_START1;
    parser->buffer = buffer;
    parser->buffer_size = buffer_size;
    parser->payload_len = 0;
    parser->received_len = 0;
    parser->text_buffer = text_buffer;
    parser->text_buffer_size = text_buffer_size;
    parser->text_len = 0;
}

frame_result_t frame_parser_feed(frame_parser_t *parser, uint8_t byte,
                                  uint8_t **payload_out, uint16_t *len_out) {
    switch (parser->state) {
        /*
         * 状态: 等待 START1 (0x94)
         * 输入: 任意字节
         * 动作:
         *   - 若收到 0x94 → 进入 WAIT_START2
         *   - 其他 → 存入文本缓冲区
         */
        case FRAME_STATE_WAIT_START1:
            if (byte == FRAME_START1) {
                parser->state = FRAME_STATE_WAIT_START2;
            } else {
                /* 非帧起始字节，视为设备输出的文本 */
                if (parser->text_len < parser->text_buffer_size) {
                    parser->text_buffer[parser->text_len++] = byte;
                }
            }
            return FRAME_RESULT_NEED_MORE;
            
        /*
         * 状态: 等待 START2 (0xC3)
         * 输入: 任意字节
         * 动作:
         *   - 若收到 0xC3 → 进入 WAIT_LEN1 (帧同步成功)
         *   - 其他 → 0x94 和当前字节都视为文本，回到 WAIT_START1
         */
        case FRAME_STATE_WAIT_START2:
            if (byte == FRAME_START2) {
                parser->state = FRAME_STATE_WAIT_LEN1;
                /* 帧开始前，刷出文本缓冲区 */
                /* 上层应检查 text_len 是否 > 0 来获取文本 */
            } else {
                /* 0x94 后不是 0xC3，两个字节都视为文本 */
                if (parser->text_len < parser->text_buffer_size) {
                    parser->text_buffer[parser->text_len++] = FRAME_START1;
                }
                if (parser->text_len < parser->text_buffer_size) {
                    parser->text_buffer[parser->text_len++] = byte;
                }
                parser->state = FRAME_STATE_WAIT_START1;
            }
            return FRAME_RESULT_NEED_MORE;
            
        /*
         * 状态: 等待长度高字节
         * 输入: 长度高字节
         * 动作: 存储长度高字节，进入 WAIT_LEN2
         */
        case FRAME_STATE_WAIT_LEN1:
            parser->payload_len = ((uint16_t)byte) << 8;
            parser->state = FRAME_STATE_WAIT_LEN2;
            return FRAME_RESULT_NEED_MORE;
            
        /*
         * 状态: 等待长度低字节
         * 输入: 长度低字节
         * 动作:
         *   - 计算完整长度
         *   - 校验长度是否有效
         *   - 有效 → 进入 WAIT_PAYLOAD
         *   - 无效 → 错误，回到 WAIT_START1
         */
        case FRAME_STATE_WAIT_LEN2:
            parser->payload_len |= byte;
            parser->received_len = 0;
            
            /* 长度校验 */
            if (parser->payload_len == 0) {
                /* 长度为0，无效帧 */
                parser->state = FRAME_STATE_WAIT_START1;
                return FRAME_RESULT_ERROR;
            }
            if (parser->payload_len > parser->buffer_size) {
                /* 长度超出缓冲区 */
                parser->state = FRAME_STATE_WAIT_START1;
                return FRAME_RESULT_ERROR;
            }
            if (parser->payload_len > MAX_PAYLOAD_LEN) {
                /* 长度超出协议上限 */
                parser->state = FRAME_STATE_WAIT_START1;
                return FRAME_RESULT_ERROR;
            }
            
            parser->state = FRAME_STATE_WAIT_PAYLOAD;
            return FRAME_RESULT_NEED_MORE;
            
        /*
         * 状态: 等待 Payload 数据
         * 输入: payload字节
         * 动作:
         *   - 存入缓冲区
         *   - 收齐后返回完成
         */
        case FRAME_STATE_WAIT_PAYLOAD:
            parser->buffer[parser->received_len++] = byte;
            
            if (parser->received_len >= parser->payload_len) {
                /* 帧接收完成 */
                *payload_out = parser->buffer;
                *len_out = parser->payload_len;
                parser->state = FRAME_STATE_WAIT_START1;
                return FRAME_RESULT_COMPLETE;
            }
            return FRAME_RESULT_NEED_MORE;
    }
    
    return FRAME_RESULT_NEED_MORE;
}

const uint8_t* frame_parser_get_text(frame_parser_t *parser, uint16_t *len_out) {
    *len_out = parser->text_len;
    return parser->text_buffer;
}

void frame_parser_clear_text(frame_parser_t *parser) {
    parser->text_len = 0;
}

void frame_parser_reset(frame_parser_t *parser) {
    parser->state = FRAME_STATE_WAIT_START1;
    parser->payload_len = 0;
    parser->received_len = 0;
    parser->text_len = 0;
}
```

## 4. 帧发送

### 4.1 接口定义

```c
// frame_builder.h
#ifndef FRAME_BUILDER_H
#define FRAME_BUILDER_H

#include <stdint.h>
#include <stdbool.h>

/**
 * 写入函数类型
 * 
 * @param data      要写入的数据
 * @param len       数据长度
 * @param user_data 用户数据
 * @return 实际写入的字节数，-1表示错误
 */
typedef int (*frame_write_fn)(const uint8_t *data, uint16_t len, void *user_data);

/**
 * 构造帧到缓冲区
 * 
 * @param out_buf    输出缓冲区 (至少 len + 4 字节)
 * @param out_size   输出缓冲区大小
 * @param payload    Payload数据
 * @param len        Payload长度
 * @return 帧总长度 (包括header)，0表示失败
 */
uint16_t frame_build(uint8_t *out_buf, uint16_t out_size, 
                      const uint8_t *payload, uint16_t len);

/**
 * 构造并发送一个帧
 * 
 * @param payload    Payload数据 (FromRadio/ToRadio序列化后的字节)
 * @param len        Payload长度
 * @param write_fn   写入函数指针
 * @param user_data  用户数据, 传递给write_fn
 * @return true=成功, false=失败
 */
bool frame_send(const uint8_t *payload, uint16_t len,
                frame_write_fn write_fn, void *user_data);

/**
 * 发送唤醒序列
 * 发送32个 0xC3 字节，唤醒休眠中的设备
 * 
 * @param write_fn   写入函数
 * @param user_data  用户数据
 * @return true=成功
 */
bool frame_send_wakeup(frame_write_fn write_fn, void *user_data);

#endif /* FRAME_BUILDER_H */
```

### 4.2 实现代码

```c
// frame_builder.c
#include "frame_builder.h"
#include "frame_protocol.h"
#include <string.h>

uint16_t frame_build(uint8_t *out_buf, uint16_t out_size,
                      const uint8_t *payload, uint16_t len) {
    /* 参数校验 */
    if (out_buf == NULL || len == 0 || len > MAX_TO_FROM_RADIO) {
        return 0;
    }
    
    uint16_t frame_len = len + FRAME_HEADER_LEN;
    if (frame_len > out_size) {
        return 0;
    }
    
    /* 构造帧头 */
    out_buf[0] = FRAME_START1;
    out_buf[1] = FRAME_START2;
    out_buf[2] = (len >> 8) & 0xFF;
    out_buf[3] = len & 0xFF;
    
    /* 复制payload */
    if (payload != NULL && len > 0) {
        memcpy(out_buf + FRAME_HEADER_LEN, payload, len);
    }
    
    return frame_len;
}

bool frame_send(const uint8_t *payload, uint16_t len,
                frame_write_fn write_fn, void *user_data) {
    if (write_fn == NULL) {
        return false;
    }
    
    if (len == 0 || len > MAX_TO_FROM_RADIO) {
        return false;
    }
    
    /* 构造帧头 */
    uint8_t header[FRAME_HEADER_LEN];
    header[0] = FRAME_START1;
    header[1] = FRAME_START2;
    header[2] = (len >> 8) & 0xFF;
    header[3] = len & 0xFF;
    
    /* 发送帧头 */
    if (write_fn(header, FRAME_HEADER_LEN, user_data) != FRAME_HEADER_LEN) {
        return false;
    }
    
    /* 发送payload */
    if (payload != NULL && len > 0) {
        if (write_fn(payload, len, user_data) != (int)len) {
            return false;
        }
    }
    
    return true;
}

bool frame_send_wakeup(frame_write_fn write_fn, void *user_data) {
    if (write_fn == NULL) {
        return false;
    }
    
    /* 32个 0xC3 字节唤醒设备 */
    uint8_t wakeup[32];
    memset(wakeup, FRAME_START2, sizeof(wakeup));
    
    return write_fn(wakeup, sizeof(wakeup), user_data) == sizeof(wakeup);
}
```

## 5. 非帧数据处理

### 5.1 文本日志处理

设备可能通过 `Serial.println()` 输出调试日志，这些不是 protobuf 帧。帧解析器已经将非帧字节存入 `text_buffer`，上层需要进一步处理：

```c
// text_handler.h
#ifndef TEXT_HANDLER_H
#define TEXT_HANDLER_H

#include <stdint.h>
#include <stdbool.h>

/* 文本行回调函数类型 */
typedef void (*text_line_callback_t)(const char *line, uint16_t len, void *user_data);

/* 文本处理器结构体 */
typedef struct {
    char *buffer;
    uint16_t buffer_size;
    uint16_t len;
    text_line_callback_t callback;
    void *user_data;
} text_handler_t;

/**
 * 初始化文本处理器
 */
void text_handler_init(text_handler_t *handler, char *buffer, uint16_t size,
                       text_line_callback_t callback, void *user_data);

/**
 * 添加数据到缓冲区并处理完整行
 * 
 * @param data  原始数据 (可能包含多行)
 * @param len   数据长度
 */
void text_handler_feed(text_handler_t *handler, const uint8_t *data, uint16_t len);

/**
 * 刷新缓冲区中剩余的数据 (作为最后一行)
 */
void text_handler_flush(text_handler_t *handler);

#endif /* TEXT_HANDLER_H */
```

```c
// text_handler.c
#include "text_handler.h"
#include <string.h>

void text_handler_init(text_handler_t *handler, char *buffer, uint16_t size,
                       text_line_callback_t callback, void *user_data) {
    handler->buffer = buffer;
    handler->buffer_size = size;
    handler->len = 0;
    handler->callback = callback;
    handler->user_data = user_data;
}

void text_handler_feed(text_handler_t *handler, const uint8_t *data, uint16_t len) {
    for (uint16_t i = 0; i < len; i++) {
        uint8_t byte = data[i];
        
        if (byte == '\r') {
            /* 忽略回车 */
            continue;
        }
        
        if (byte == '\n') {
            /* 完整一行 */
            if (handler->len > 0 && handler->callback != NULL) {
                handler->buffer[handler->len] = '\0';
                handler->callback(handler->buffer, handler->len, handler->user_data);
            }
            handler->len = 0;
            continue;
        }
        
        /* 添加到缓冲区 */
        if (handler->len < handler->buffer_size - 1) {
            handler->buffer[handler->len++] = (char)byte;
        }
    }
}

void text_handler_flush(text_handler_t *handler) {
    if (handler->len > 0 && handler->callback != NULL) {
        handler->buffer[handler->len] = '\0';
        handler->callback(handler->buffer, handler->len, handler->user_data);
    }
    handler->len = 0;
}
```

## 6. 握手协议

### 6.1 连接建立流程

```
主机                                    设备
  │                                       │
  │──── 发送唤醒序列 (32 × 0xC3) ────────>│  (可选，唤醒休眠设备)
  │                                       │
  │──── 发送 want_config_id ────────────>│
  │                                       │
  │<─── my_info ────────────────────────│
  │<─── node_info (多个) ───────────────│
  │<─── config ─────────────────────────│
  │<─── channel (多个) ─────────────────│
  │<─── moduleConfig ──────────────────│
  │<─── metadata (可选) ───────────────│
  │<─── config_complete_id ────────────│
  │                                       │
  │        配置同步完成                    │
```

### 6.2 want_config_id 构造

```c
/**
 * ToRadio.want_config_id 消息结构:
 * 
 * Protobuf定义:
 *   message ToRadio {
 *       fixed32 want_config_id = 100;
 *   }
 * 
 * 编码格式:
 *   Tag: field_number=100, wire_type=5 (32-bit fixed)
 *        tag = (100 << 3) | 5 = 0x3205
 *        编码为: 0x85 0xC2 0x04 (varint编码)
 *   Value: 4字节 fixed32 (小端)
 */

bool send_want_config(frame_write_fn write_fn, void *user_data, uint32_t config_id) {
    /* 
     * ToRadio.want_config_id 的 protobuf 编码
     * Tag = (field_number << 3) | wire_type
     *     = (100 << 3) | 5 = 805 = 0x0325
     *     varint编码: 0xA5 0x06
     */
    uint8_t payload[8];
    uint8_t len = 0;
    
    /* Tag (varint) */
    payload[len++] = 0xA5;  /* (100 << 3) | 5 = 805, 低7位 = 0x25 | 0x80 */
    payload[len++] = 0x06;  /* 高位 = 6 */
    
    /* Value (fixed32, 小端) */
    payload[len++] = config_id & 0xFF;
    payload[len++] = (config_id >> 8) & 0xFF;
    payload[len++] = (config_id >> 16) & 0xFF;
    payload[len++] = (config_id >> 24) & 0xFF;
    
    return frame_send(payload, len, write_fn, user_data);
}

/* 使用 nanopb 库的版本 */
#include "mesh.pb.h"

bool send_want_config_pb(frame_write_fn write_fn, void *user_data, uint32_t config_id) {
    meshtastic_ToRadio to_radio = meshtastic_ToRadio_init_zero;
    to_radio.which_payload_variant = meshtastic_ToRadio_want_config_id_tag;
    to_radio.want_config_id = config_id;
    
    uint8_t buffer[16];
    pb_ostream_t stream = pb_ostream_from_buffer(buffer, sizeof(buffer));
    
    if (!pb_encode(&stream, meshtastic_ToRadio_fields, &to_radio)) {
        return false;
    }
    
    return frame_send(buffer, stream.bytes_written, write_fn, user_data);
}
```

### 6.3 设备响应序列

发送 `want_config_id` 后，设备会依次发送以下 FromRadio 消息：

| 序号 | variant | 消息类型 | 说明 |
|------|---------|----------|------|
| 1 | my_info | MyNodeInfo | 本机节点信息 |
| 2 | node_info | NodeInfo | 已知节点信息 (可能有多个) |
| 3 | config | Config | 设备配置 |
| 4 | channel | Channel | 信道配置 (可能有多个) |
| 5 | moduleConfig | ModuleConfig | 模块配置 |
| 6 | metadata | DeviceMetadata | 固件元数据 (可选) |
| 7 | config_complete_id | uint32 | 配置完成标志 |

**config_complete_id 的值等于请求时发送的 config_id，用于确认同步完成。**

## 7. 完整接收示例

```c
/* main.c - 完整接收循环示例 */

#include "frame_protocol.h"
#include "frame_parser.h"
#include "frame_builder.h"
#include "text_handler.h"
#include "from_radio_parser.h"  /* 下一个文档 */
#include "serial_port.h"

/* 缓冲区 */
static uint8_t frame_buffer[2048];
static char text_buffer[4096];

/* 解析器 */
static frame_parser_t parser;
static text_handler_t text_handler;

/* 运行标志 */
static volatile bool running = true;

/* 文本日志回调 */
static void on_text_line(const char *line, uint16_t len, void *user_data) {
    (void)user_data;
    printf("[LOG] %.*s\n", len, line);
}

/* 帧处理回调 */
static void on_frame_received(const uint8_t *payload, uint16_t len) {
    FromRadio from_radio;
    
    if (from_radio_parse(payload, len, &from_radio)) {
        switch (from_radio.which_payload_variant) {
            case FromRadio_packet_tag:
                handle_mesh_packet(&from_radio.packet);
                break;
                
            case FromRadio_my_info_tag:
                handle_my_info(&from_radio.my_info);
                break;
                
            case FromRadio_node_info_tag:
                handle_node_info(&from_radio.node_info);
                break;
                
            case FromRadio_config_tag:
                handle_config(&from_radio.config);
                break;
                
            case FromRadio_channel_tag:
                handle_channel(&from_radio.channel);
                break;
                
            case FromRadio_moduleConfig_tag:
                handle_module_config(&from_radio.moduleConfig);
                break;
                
            case FromRadio_config_complete_id_tag:
                printf("Config sync complete, id=%u\n", from_radio.config_complete_id);
                /* 配置同步完成，可以开始正常工作 */
                break;
                
            case FromRadio_rebooted_tag:
                printf("Device rebooted, re-syncing...\n");
                send_want_config(serial_write, &serial_port, generate_config_id());
                break;
                
            default:
                printf("Unknown variant: %d\n", from_radio.which_payload_variant);
                break;
        }
    }
}

/* 主接收循环 */
void receive_loop(serial_port_t *port) {
    /* 初始化解析器 */
    frame_parser_init(&parser, frame_buffer, sizeof(frame_buffer),
                      (uint8_t*)text_buffer, sizeof(text_buffer));
    
    /* 初始化文本处理器 */
    text_handler_init(&text_handler, text_buffer, sizeof(text_buffer),
                      on_text_line, NULL);
    
    /* 发送握手请求 */
    send_wakeup(serial_write, port);
    delay_ms(100);
    send_want_config(serial_write, port, generate_config_id());
    
    /* 主循环 */
    while (running) {
        uint8_t byte;
        
        if (serial_read_byte(port, &byte, READ_TIMEOUT_MS)) {
            uint8_t *payload;
            uint16_t len;
            
            frame_result_t result = frame_parser_feed(&parser, byte, &payload, &len);
            
            if (result == FRAME_RESULT_COMPLETE) {
                /* 帧接收完成 */
                on_frame_received(payload, len);
                
                /* 处理缓冲的文本数据 */
                uint16_t text_len;
                const uint8_t *text = frame_parser_get_text(&parser, &text_len);
                if (text_len > 0) {
                    text_handler_feed(&text_handler, text, text_len);
                    frame_parser_clear_text(&parser);
                }
            } else if (result == FRAME_RESULT_ERROR) {
                printf("Frame parse error\n");
            }
        }
    }
}
```

### 8.2 重连机制

```c
/* 重连配置 */
typedef struct {
    int max_retries;          /* 最大重试次数 */
    int initial_delay_ms;     /* 初始延迟 (毫秒) */
    int max_delay_ms;         /* 最大延迟 (毫秒) */
    int current_retry;
    int current_delay_ms;
} reconnect_config_t;

void reconnect_config_init(reconnect_config_t *cfg) {
    cfg->max_retries = 5;
    cfg->initial_delay_ms = 1000;
    cfg->max_delay_ms = 30000;
    cfg->current_retry = 0;
    cfg->current_delay_ms = cfg->initial_delay_ms;
}

bool reconnect_attempt(reconnect_config_t *cfg, serial_port_t *port) {
    if (cfg->current_retry >= cfg->max_retries) {
        return false;  /* 放弃重连 */
    }
    
    cfg->current_retry++;
    
    /* 关闭旧连接 */
    serial_close(port);
    
    /* 等待 */
    delay_ms(cfg->current_delay_ms);
    
    /* 尝试重连 */
    if (serial_open(port)) {
        /* 重连成功，重置配置 */
        cfg->current_retry = 0;
        cfg->current_delay_ms = cfg->initial_delay_ms;
        return true;
    }
    
    /* 指数退避 */
    cfg->current_delay_ms *= 2;
    if (cfg->current_delay_ms > cfg->max_delay_ms) {
        cfg->current_delay_ms = cfg->max_delay_ms;
    }
    
    return true;  /* 继续重试 */
}
```

### 9.2 Windows 实现

```c
/* serial_port_win.c */
#ifdef _WIN32

#include "serial_port.h"
#include <windows.h>
#include <setupapi.h>
#include <initguid.h>
#include <devguid.h>

struct serial_port {
    HANDLE handle;
};

serial_port_t* serial_open(const char *port, uint32_t baudrate) {
    serial_port_t *sp = malloc(sizeof(serial_port_t));
    if (!sp) return NULL;
    
    /* 构造完整路径 */
    char path[64];
    if (port[0] != '\\') {
        snprintf(path, sizeof(path), "\\\\.\\%s", port);
    } else {
        strncpy(path, port, sizeof(path) - 1);
    }
    
    /* 打开串口 */
    sp->handle = CreateFileA(
        path,
        GENERIC_READ | GENERIC_WRITE,
        0, NULL, OPEN_EXISTING,
        FILE_ATTRIBUTE_NORMAL, NULL
    );
    
    if (sp->handle == INVALID_HANDLE_VALUE) {
        free(sp);
        return NULL;
    }
    
    /* 配置串口参数 */
    DCB dcb = {0};
    dcb.DCBlength = sizeof(DCB);
    GetCommState(sp->handle, &dcb);
    
    dcb.BaudRate = baudrate;
    dcb.ByteSize = 8;
    dcb.Parity = NOPARITY;
    dcb.StopBits = ONESTOPBIT;
    
    SetCommState(sp->handle, &dcb);
    
    /* 设置超时 */
    COMMTIMEOUTS timeouts = {0};
    timeouts.ReadIntervalTimeout = MAXDWORD;
    timeouts.ReadTotalTimeoutMultiplier = 0;
    timeouts.ReadTotalTimeoutConstant = 0;
    SetCommTimeouts(sp->handle, &timeouts);
    
    return sp;
}

void serial_close(serial_port_t *sp) {
    if (sp) {
        if (sp->handle != INVALID_HANDLE_VALUE) {
            CloseHandle(sp->handle);
        }
        free(sp);
    }
}

bool serial_read_byte(serial_port_t *sp, uint8_t *byte, uint32_t timeout_ms) {
    if (!sp || sp->handle == INVALID_HANDLE_VALUE) {
        return false;
    }
    
    /* 设置临时超时 */
    COMMTIMEOUTS timeouts = {0};
    timeouts.ReadIntervalTimeout = MAXDWORD;
    timeouts.ReadTotalTimeoutMultiplier = 0;
    timeouts.ReadTotalTimeoutConstant = timeout_ms;
    SetCommTimeouts(sp->handle, &timeouts);
    
    DWORD read;
    if (ReadFile(sp->handle, byte, 1, &read, NULL) && read == 1) {
        return true;
    }
    return false;
}

int serial_write(const uint8_t *data, uint16_t len, void *user_data) {
    serial_port_t *sp = (serial_port_t*)user_data;
    if (!sp || sp->handle == INVALID_HANDLE_VALUE) {
        return -1;
    }
    
    DWORD written;
    if (WriteFile(sp->handle, data, len, &written, NULL)) {
        FlushFileBuffers(sp->handle);
        return (int)written;
    }
    return -1;
}

#endif /* _WIN32 */
```
