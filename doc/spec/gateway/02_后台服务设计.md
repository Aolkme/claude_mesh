# 02_后台服务设计

> 本文档详细说明 meshgateway 后台守护进程的设计与实现。

## 1. 守护进程概述

### 1.1 进程特性

| 特性 | 说明 |
|------|------|
| 类型 | systemd notify 型守护进程 |
| 用户 | meshgate (非 root) |
| 工作目录 | /opt/meshgateway |
| PID文件 | /run/meshgateway.pid |
| 标准输出 | journal (systemd日志) |

### 1.2 进程生命周期

```
启动阶段:
  main() → 解析参数 → 加载配置 → 初始化日志
        → 创建PID文件 → 初始化各模块 → 通知systemd就绪
        → 进入主循环

运行阶段:
  主循环 → 处理信号 → 处理定时器 → 处理事件
        → epoll_wait → 处理I/O

关闭阶段:
  收到SIGTERM → 设置退出标志 → 等待线程退出
              → 保存状态 → 清理资源 → 退出
```

## 2. 主程序结构

### 2.1 主函数框架

```c
// main.c
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <unistd.h>
#include <systemd/sd-daemon.h>

#include "config.h"
#include "log.h"
#include "event_loop.h"
#include "serial_manager.h"
#include "socket_server.h"
#include "node_manager.h"

/* 全局状态 */
static volatile int g_running = 1;
static config_t g_config;
static event_loop_t g_loop;

/* 信号处理 */
static void signal_handler(int sig) {
    switch (sig) {
        case SIGTERM:
        case SIGINT:
            g_running = 0;
            break;
        case SIGHUP:
            /* 重载配置 */
            config_load(&g_config, CONFIG_PATH);
            log_info("main", "Configuration reloaded");
            break;
        case SIGUSR1:
            /* 调试信息 */
            node_manager_dump_stats();
            break;
    }
}

/* 初始化信号处理 */
static int setup_signals(void) {
    struct sigaction sa = {0};
    sa.sa_handler = signal_handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = 0;
    
    sigaction(SIGTERM, &sa, NULL);
    sigaction(SIGINT, &sa, NULL);
    sigaction(SIGHUP, &sa, NULL);
    sigaction(SIGUSR1, &sa, NULL);
    
    /* 忽略 SIGPIPE */
    signal(SIGPIPE, SIG_IGN);
    
    return 0;
}

/* 创建 PID 文件 */
static int create_pid_file(void) {
    FILE *fp = fopen(PID_PATH, "w");
    if (!fp) return -1;
    fprintf(fp, "%d\n", getpid());
    fclose(fp);
    return 0;
}

/* 初始化 */
static int init(void) {
    /* 1. 加载配置 */
    if (config_load(&g_config, CONFIG_PATH) != 0) {
        fprintf(stderr, "Failed to load config\n");
        return -1;
    }
    
    /* 2. 初始化日志 */
    if (log_init(g_config.log_file, g_config.log_level,
                 g_config.log_max_size, g_config.log_max_files) != 0) {
        fprintf(stderr, "Failed to init log\n");
        return -1;
    }
    
    log_info("main", "MeshGateway starting...");
    
    /* 3. 创建 PID 文件 */
    if (create_pid_file() != 0) {
        log_error("main", "Failed to create PID file");
        return -1;
    }
    
    /* 4. 设置信号 */
    if (setup_signals() != 0) {
        log_error("main", "Failed to setup signals");
        return -1;
    }
    
    /* 5. 初始化事件循环 */
    if (event_loop_init(&g_loop) != 0) {
        log_error("main", "Failed to init event loop");
        return -1;
    }
    
    /* 6. 初始化节点管理器 */
    if (node_manager_init() != 0) {
        log_error("main", "Failed to init node manager");
        return -1;
    }
    
    /* 7. 初始化串口管理器 */
    if (serial_manager_init(&g_config) != 0) {
        log_error("main", "Failed to init serial manager");
        return -1;
    }
    
    /* 8. 初始化 Socket 服务器 */
    if (socket_server_init(&g_config) != 0) {
        log_error("main", "Failed to init socket server");
        return -1;
    }
    
    return 0;
}

/* 清理 */
static void cleanup(void) {
    log_info("main", "MeshGateway stopping...");
    
    socket_server_close();
    serial_manager_close();
    node_manager_close();
    event_loop_close(&g_loop);
    
    unlink(PID_PATH);
    log_close();
}

/* 主函数 */
int main(int argc, char *argv[]) {
    /* 初始化 */
    if (init() != 0) {
        cleanup();
        return 1;
    }
    
    /* 通知 systemd 就绪 */
    sd_notify(0, "READY=1");
    log_info("main", "Service ready");
    
    /* 主循环 */
    while (g_running) {
        event_loop_run(&g_loop, 1000);  /* 1秒超时 */
        
        /* 定时任务 */
        static uint64_t last_check = 0;
        uint64_t now = get_time_ms();
        
        /* 心跳检查 (每秒) */
        if (now - last_check >= 1000) {
            heartbeat_tick();
            node_manager_check_expired();
            last_check = now;
        }
    }
    
    /* 通知 systemd 停止 */
    sd_notify(0, "STOPPING=1");
    
    /* 清理 */
    cleanup();
    
    return 0;
}
```

## 3. 事件循环设计

### 3.1 事件类型

```c
// event_loop.h
#ifndef EVENT_LOOP_H
#define EVENT_LOOP_H

#include <stdint.h>

/* 事件类型 */
typedef enum {
    EVENT_TYPE_TIMER,       /* 定时器事件 */
    EVENT_TYPE_IO,          /* I/O事件 */
    EVENT_TYPE_SIGNAL,      /* 信号事件 */
    EVENT_TYPE_CUSTOM,      /* 自定义事件 */
} event_type_t;

/* 事件回调 */
typedef void (*event_callback_t)(int fd, uint32_t events, void *data);

/* 定时器回调 */
typedef void (*timer_callback_t)(void *data);

/* I/O 事件 */
typedef struct {
    int fd;
    uint32_t events;        /* EPOLLIN, EPOLLOUT, etc. */
    event_callback_t callback;
    void *data;
} io_event_t;

/* 定时器事件 */
typedef struct {
    uint64_t interval_ms;
    uint64_t next_trigger;
    timer_callback_t callback;
    void *data;
    int enabled;
} timer_event_t;

/* 事件循环 */
typedef struct {
    int epoll_fd;
    io_event_t *io_events;
    int io_count;
    int io_capacity;
    timer_event_t *timers;
    int timer_count;
    int timer_capacity;
} event_loop_t;

/* 初始化 */
int event_loop_init(event_loop_t *loop);

/* 关闭 */
void event_loop_close(event_loop_t *loop);

/* 运行 (阻塞) */
int event_loop_run(event_loop_t *loop, int timeout_ms);

/* 添加 I/O 事件 */
int event_loop_add_io(event_loop_t *loop, int fd, uint32_t events,
                       event_callback_t callback, void *data);

/* 删除 I/O 事件 */
int event_loop_del_io(event_loop_t *loop, int fd);

/* 添加定时器 */
int event_loop_add_timer(event_loop_t *loop, uint64_t interval_ms,
                          timer_callback_t callback, void *data);

/* 删除定时器 */
int event_loop_del_timer(event_loop_t *loop, timer_callback_t callback);

#endif /* EVENT_LOOP_H */
```

### 3.2 事件循环实现

```c
// event_loop.c
#include "event_loop.h"
#include <stdlib.h>
#include <string.h>
#include <sys/epoll.h>
#include <unistd.h>
#include <time.h>

#define MAX_EVENTS 64

static uint64_t get_time_ms(void) {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (uint64_t)ts.tv_sec * 1000 + ts.tv_nsec / 1000000;
}

int event_loop_init(event_loop_t *loop) {
    memset(loop, 0, sizeof(event_loop_t));
    
    loop->epoll_fd = epoll_create1(0);
    if (loop->epoll_fd < 0) {
        return -1;
    }
    
    loop->io_capacity = 32;
    loop->io_events = calloc(loop->io_capacity, sizeof(io_event_t));
    
    loop->timer_capacity = 16;
    loop->timers = calloc(loop->timer_capacity, sizeof(timer_event_t));
    
    return 0;
}

void event_loop_close(event_loop_t *loop) {
    if (loop->epoll_fd >= 0) {
        close(loop->epoll_fd);
    }
    free(loop->io_events);
    free(loop->timers);
}

int event_loop_add_io(event_loop_t *loop, int fd, uint32_t events,
                       event_callback_t callback, void *data) {
    /* 扩容检查 */
    if (loop->io_count >= loop->io_capacity) {
        loop->io_capacity *= 2;
        loop->io_events = realloc(loop->io_events, 
                                   loop->io_capacity * sizeof(io_event_t));
    }
    
    /* 添加到 epoll */
    struct epoll_event ev = {0};
    ev.events = events;
    ev.data.fd = fd;
    
    if (epoll_ctl(loop->epoll_fd, EPOLL_CTL_ADD, fd, &ev) < 0) {
        return -1;
    }
    
    /* 记录事件 */
    io_event_t *io = &loop->io_events[loop->io_count++];
    io->fd = fd;
    io->events = events;
    io->callback = callback;
    io->data = data;
    
    return 0;
}

int event_loop_del_io(event_loop_t *loop, int fd) {
    epoll_ctl(loop->epoll_fd, EPOLL_CTL_DEL, fd, NULL);
    
    for (int i = 0; i < loop->io_count; i++) {
        if (loop->io_events[i].fd == fd) {
            /* 移动最后一个到当前位置 */
            if (i < loop->io_count - 1) {
                loop->io_events[i] = loop->io_events[loop->io_count - 1];
            }
            loop->io_count--;
            return 0;
        }
    }
    return -1;
}

int event_loop_add_timer(event_loop_t *loop, uint64_t interval_ms,
                          timer_callback_t callback, void *data) {
    if (loop->timer_count >= loop->timer_capacity) {
        loop->timer_capacity *= 2;
        loop->timers = realloc(loop->timers,
                                loop->timer_capacity * sizeof(timer_event_t));
    }
    
    timer_event_t *timer = &loop->timers[loop->timer_count++];
    timer->interval_ms = interval_ms;
    timer->next_trigger = get_time_ms() + interval_ms;
    timer->callback = callback;
    timer->data = data;
    timer->enabled = 1;
    
    return 0;
}

int event_loop_run(event_loop_t *loop, int timeout_ms) {
    struct epoll_event events[MAX_EVENTS];
    uint64_t now = get_time_ms();
    
    /* 1. 处理定时器 */
    for (int i = 0; i < loop->timer_count; i++) {
        timer_event_t *timer = &loop->timers[i];
        if (timer->enabled && now >= timer->next_trigger) {
            timer->callback(timer->data);
            timer->next_trigger = now + timer->interval_ms;
        }
    }
    
    /* 2. 计算 epoll 超时 */
    int min_timeout = timeout_ms;
    for (int i = 0; i < loop->timer_count; i++) {
        timer_event_t *timer = &loop->timers[i];
        if (timer->enabled) {
            int remaining = (int)(timer->next_trigger - now);
            if (remaining < min_timeout) {
                min_timeout = remaining;
            }
        }
    }
    if (min_timeout < 0) min_timeout = 0;
    
    /* 3. epoll_wait */
    int n = epoll_wait(loop->epoll_fd, events, MAX_EVENTS, min_timeout);
    if (n < 0) {
        return -1;
    }
    
    /* 4. 处理 I/O 事件 */
    for (int i = 0; i < n; i++) {
        int fd = events[i].data.fd;
        
        /* 查找回调 */
        for (int j = 0; j < loop->io_count; j++) {
            if (loop->io_events[j].fd == fd) {
                loop->io_events[j].callback(fd, events[i].events,
                                             loop->io_events[j].data);
                break;
            }
        }
    }
    
    return 0;
}
```

### 4.2 串口线程

```c
// serial_manager.c
#include "serial_manager.h"
#include "frame_parser.h"
#include "log.h"
#include <pthread.h>
#include <fcntl.h>
#include <termios.h>
#include <unistd.h>
#include <string.h>
#include <errno.h>

static struct {
    int fd;
    serial_state_t state;
    pthread_t thread;
    volatile int running;
    
    config_t config;
    frame_parser_t parser;
    
    serial_frame_callback_t frame_cb;
    serial_text_callback_t text_cb;
    serial_state_callback_t state_cb;
    
    pthread_mutex_t write_lock;
} g_serial;

/* 设置串口参数 */
static int setup_serial(int fd, uint32_t baudrate) {
    struct termios tty;
    if (tcgetattr(fd, &tty) != 0) return -1;
    
    speed_t speed = B115200;
    switch (baudrate) {
        case 9600:   speed = B9600; break;
        case 19200:  speed = B19200; break;
        case 38400:  speed = B38400; break;
        case 57600:  speed = B57600; break;
        case 115200: speed = B115200; break;
        case 230400: speed = B230400; break;
    }
    
    cfsetospeed(&tty, speed);
    cfsetispeed(&tty, speed);
    
    tty.c_cflag &= ~PARENB;     /* 无校验 */
    tty.c_cflag &= ~CSTOPB;     /* 1停止位 */
    tty.c_cflag &= ~CSIZE;
    tty.c_cflag |= CS8;         /* 8数据位 */
    tty.c_cflag &= ~CRTSCTS;    /* 无硬件流控 */
    tty.c_cflag |= CLOCAL | CREAD;
    
    tty.c_iflag &= ~(IXON | IXOFF | IXANY);
    tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL);
    
    tty.c_oflag &= ~OPOST;
    tty.c_lflag &= ~(ICANON | ECHO | ECHOE | ISIG);
    
    tty.c_cc[VMIN] = 1;
    tty.c_cc[VTIME] = 0;
    
    if (tcsetattr(fd, TCSANOW, &tty) != 0) return -1;
    
    return 0;
}

/* 设置状态 */
static void set_state(serial_state_t state) {
    if (g_serial.state != state) {
        g_serial.state = state;
        if (g_serial.state_cb) {
            g_serial.state_cb(state);
        }
    }
}

/* 串口线程 */
static void* serial_thread(void *arg) {
    uint8_t frame_buf[2048];
    uint8_t text_buf[4096];
    uint8_t byte;
    
    frame_parser_init(&g_serial.parser, frame_buf, sizeof(frame_buf),
                      text_buf, sizeof(text_buf));
    
    while (g_serial.running) {
        /* 1. 检查连接状态 */
        if (g_serial.fd < 0) {
            /* 尝试重连 */
            g_serial.fd = open(g_serial.config.serial_device, O_RDWR | O_NOCTTY);
            if (g_serial.fd >= 0) {
                if (setup_serial(g_serial.fd, g_serial.config.baudrate) == 0) {
                    set_state(SERIAL_STATE_CONNECTED);
                    log_info("serial", "Connected to %s", 
                             g_serial.config.serial_device);
                } else {
                    close(g_serial.fd);
                    g_serial.fd = -1;
                }
            }
            
            if (g_serial.fd < 0) {
                sleep(g_serial.config.reconnect_interval);
                continue;
            }
        }
        
        /* 2. 读取数据 */
        ssize_t n = read(g_serial.fd, &byte, 1);
        if (n <= 0) {
            if (n < 0 && errno != EAGAIN) {
                log_error("serial", "Read error: %s", strerror(errno));
                close(g_serial.fd);
                g_serial.fd = -1;
                set_state(SERIAL_STATE_ERROR);
            }
            continue;
        }
        
        /* 3. 帧解析 */
        uint8_t *payload;
        uint16_t len;
        
        int result = frame_parser_feed(&g_serial.parser, byte, &payload, &len);
        
        if (result == FRAME_RESULT_COMPLETE) {
            /* 帧完成 */
            if (g_serial.frame_cb) {
                g_serial.frame_cb(payload, len);
            }
        }
        
        /* 4. 处理文本数据 */
        uint16_t text_len;
        const uint8_t *text = frame_parser_get_text(&g_serial.parser, &text_len);
        if (text_len > 0 && g_serial.text_cb) {
            g_serial.text_cb((const char*)text, text_len);
            frame_parser_clear_text(&g_serial.parser);
        }
    }
    
    return NULL;
}

int serial_manager_init(const config_t *cfg) {
    memset(&g_serial, 0, sizeof(g_serial));
    memcpy(&g_serial.config, cfg, sizeof(config_t));
    
    g_serial.fd = -1;
    g_serial.state = SERIAL_STATE_DISCONNECTED;
    g_serial.running = 1;
    
    pthread_mutex_init(&g_serial.write_lock, NULL);
    
    /* 启动线程 */
    if (pthread_create(&g_serial.thread, NULL, serial_thread, NULL) != 0) {
        return -1;
    }
    
    return 0;
}

void serial_manager_close(void) {
    g_serial.running = 0;
    pthread_join(g_serial.thread, NULL);
    
    if (g_serial.fd >= 0) {
        close(g_serial.fd);
    }
    
    pthread_mutex_destroy(&g_serial.write_lock);
}

int serial_send(const uint8_t *data, uint16_t len) {
    if (g_serial.fd < 0) return -1;
    
    pthread_mutex_lock(&g_serial.write_lock);
    ssize_t n = write(g_serial.fd, data, len);
    fsync(g_serial.fd);
    pthread_mutex_unlock(&g_serial.write_lock);
    
    return (n == len) ? 0 : -1;
}

int serial_send_frame(const uint8_t *payload, uint16_t len) {
    uint8_t header[4] = {0x94, 0xC3, (len >> 8) & 0xFF, len & 0xFF};
    
    if (serial_send(header, 4) != 0) return -1;
    if (serial_send(payload, len) != 0) return -1;
    
    return 0;
}

void serial_set_frame_callback(serial_frame_callback_t cb) {
    g_serial.frame_cb = cb;
}

void serial_set_text_callback(serial_text_callback_t cb) {
    g_serial.text_cb = cb;
}

void serial_set_state_callback(serial_state_callback_t cb) {
    g_serial.state_cb = cb;
}

serial_state_t serial_get_state(void) {
    return g_serial.state;
}
```

### 5.2 Socket 实现

```c
// socket_server.c
#include "socket_server.h"
#include "command_handler.h"
#include "log.h"
#include <sys/socket.h>
#include <sys/un.h>
#include <unistd.h>
#include <string.h>
#include <stdlib.h>
#include <errno.h>

#define MAX_CLIENTS 16
#define RECV_BUF_SIZE 65536

struct client {
    int fd;
    int id;
    char recv_buf[RECV_BUF_SIZE];
    int recv_len;
    uint32_t subscriptions;  /* 位掩码 */
};

static struct {
    int server_fd;
    struct client clients[MAX_CLIENTS];
    int client_count;
    int next_client_id;
    char socket_path[256];
} g_server;

/* 查找客户端 */
static struct client* find_client(int fd) {
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_server.clients[i].fd == fd) {
            return &g_server.clients[i];
        }
    }
    return NULL;
}

/* 查找空闲槽 */
static struct client* find_free_slot(void) {
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_server.clients[i].fd < 0) {
            return &g_server.clients[i];
        }
    }
    return NULL;
}

/* 关闭客户端 */
static void close_client(struct client *c) {
    if (c->fd >= 0) {
        close(c->fd);
        c->fd = -1;
        g_server.client_count--;
        log_info("socket", "Client %d disconnected", c->id);
    }
}

/* 处理客户端数据 */
static void handle_client_data(struct client *c) {
    char buf[4096];
    ssize_t n = recv(c->fd, buf, sizeof(buf) - 1, 0);
    
    if (n <= 0) {
        close_client(c);
        return;
    }
    
    /* 追加到缓冲区 */
    if (c->recv_len + n < RECV_BUF_SIZE) {
        memcpy(c->recv_buf + c->recv_len, buf, n);
        c->recv_len += n;
    } else {
        /* 缓冲区溢出 */
        log_warn("socket", "Client %d buffer overflow", c->id);
        close_client(c);
        return;
    }
    
    /* 解析消息 (以换行分隔) */
    char *start = c->recv_buf;
    char *end;
    
    while ((end = memchr(start, '\n', c->recv_len - (start - c->recv_buf))) != NULL) {
        *end = '\0';
        
        /* 处理 JSON 消息 */
        char response[65536];
        int resp_len = command_handler_process(c->id, start, response, sizeof(response));
        
        if (resp_len > 0) {
            /* 发送响应 */
            send(c->fd, response, resp_len, 0);
            send(c->fd, "\n", 1, 0);
        }
        
        start = end + 1;
    }
    
    /* 移动剩余数据 */
    int remaining = c->recv_len - (start - c->recv_buf);
    if (remaining > 0 && start != c->recv_buf) {
        memmove(c->recv_buf, start, remaining);
    }
    c->recv_len = remaining;
}

/* 接受新连接 */
static void accept_connection(void) {
    int fd = accept(g_server.server_fd, NULL, NULL);
    if (fd < 0) {
        return;
    }
    
    if (g_server.client_count >= MAX_CLIENTS) {
        log_warn("socket", "Max clients reached, rejecting");
        close(fd);
        return;
    }
    
    struct client *c = find_free_slot();
    c->fd = fd;
    c->id = g_server.next_client_id++;
    c->recv_len = 0;
    c->subscriptions = 0;
    
    g_server.client_count++;
    
    log_info("socket", "Client %d connected (total: %d)", 
             c->id, g_server.client_count);
}

int socket_server_init(const config_t *cfg) {
    memset(&g_server, 0, sizeof(g_server));
    strncpy(g_server.socket_path, cfg->socket_path, sizeof(g_server.socket_path) - 1);
    
    /* 初始化客户端槽 */
    for (int i = 0; i < MAX_CLIENTS; i++) {
        g_server.clients[i].fd = -1;
    }
    
    /* 创建 socket */
    g_server.server_fd = socket(AF_UNIX, SOCK_STREAM, 0);
    if (g_server.server_fd < 0) {
        log_error("socket", "Failed to create socket: %s", strerror(errno));
        return -1;
    }
    
    /* 设置非阻塞 */
    int flags = fcntl(g_server.server_fd, F_GETFL, 0);
    fcntl(g_server.server_fd, F_SETFL, flags | O_NONBLOCK);
    
    /* 绑定 */
    struct sockaddr_un addr;
    memset(&addr, 0, sizeof(addr));
    addr.sun_family = AF_UNIX;
    strncpy(addr.sun_path, g_server.socket_path, sizeof(addr.sun_path) - 1);
    
    unlink(g_server.socket_path);  /* 删除旧文件 */
    
    if (bind(g_server.server_fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        log_error("socket", "Failed to bind: %s", strerror(errno));
        close(g_server.server_fd);
        return -1;
    }
    
    /* 设置权限 */
    chmod(g_server.socket_path, 0666);
    
    /* 监听 */
    if (listen(g_server.server_fd, 8) < 0) {
        log_error("socket", "Failed to listen: %s", strerror(errno));
        close(g_server.server_fd);
        unlink(g_server.socket_path);
        return -1;
    }
    
    log_info("socket", "Server listening on %s", g_server.socket_path);
    
    /* 添加到事件循环 */
    event_loop_add_io(&g_loop, g_server.server_fd, EPOLLIN, 
                       (event_callback_t)accept_connection, NULL);
    
    return 0;
}

int socket_server_broadcast(const char *event, const char *json_data) {
    char msg[65536];
    int len = snprintf(msg, sizeof(msg), 
        "{\"event\":\"%s\",\"data\":%s}\n", event, json_data);
    
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_server.clients[i].fd >= 0) {
            send(g_server.clients[i].fd, msg, len, 0);
        }
    }
    
    return 0;
}

int socket_server_send(int client_id, const char *json_data) {
    struct client *c = find_client_by_id(client_id);
    if (!c || c->fd < 0) return -1;
    
    send(c->fd, json_data, strlen(json_data), 0);
    send(c->fd, "\n", 1, 0);
    
    return 0;
}

void socket_server_close(void) {
    /* 关闭所有客户端 */
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_server.clients[i].fd >= 0) {
            close(g_server.clients[i].fd);
        }
    }
    
    /* 关闭服务器 */
    if (g_server.server_fd >= 0) {
        close(g_server.server_fd);
    }
    
    /* 删除 socket 文件 */
    unlink(g_server.socket_path);
}
```

### 6.2 节点管理实现

```c
// node_manager.c
#include "node_manager.h"
#include "log.h"
#include <string.h>
#include <stdlib.h>

static struct {
    node_t nodes[MAX_NODES];
    int count;
    uint32_t expire_time;
} g_nodes;

int node_manager_init(void) {
    memset(&g_nodes, 0, sizeof(g_nodes));
    return 0;
}

void node_manager_close(void) {
    /* 清理 */
    memset(&g_nodes, 0, sizeof(g_nodes));
}

/* 查找空闲槽 */
static int find_free_slot(void) {
    for (int i = 0; i < MAX_NODES; i++) {
        if (!g_nodes.nodes[i].is_valid) {
            return i;
        }
    }
    return -1;
}

/* 查找节点 */
static int find_node_idx(const char *node_id) {
    for (int i = 0; i < MAX_NODES; i++) {
        if (g_nodes.nodes[i].is_valid && 
            strcmp(g_nodes.nodes[i].node_id, node_id) == 0) {
            return i;
        }
    }
    return -1;
}

int node_manager_update(const node_t *node) {
    if (!node || !node->node_id[0]) return -1;
    
    int idx = find_node_idx(node->node_id);
    
    if (idx < 0) {
        /* 新节点 */
        idx = find_free_slot();
        if (idx < 0) {
            log_warn("node", "Node database full");
            return -1;
        }
        
        memcpy(&g_nodes.nodes[idx], node, sizeof(node_t));
        g_nodes.nodes[idx].first_seen = get_time_ms();
        g_nodes.nodes[idx].is_valid = true;
        g_nodes.count++;
        
        log_info("node", "New node: %s (%s)", 
                 node->node_id, node->long_name);
    } else {
        /* 更新节点 */
        uint64_t first_seen = g_nodes.nodes[idx].first_seen;
        memcpy(&g_nodes.nodes[idx], node, sizeof(node_t));
        g_nodes.nodes[idx].first_seen = first_seen;
        g_nodes.nodes[idx].is_valid = true;
    }
    
    g_nodes.nodes[idx].last_heard = get_time_ms();
    
    return 0;
}

node_t* node_manager_get(const char *node_id) {
    int idx = find_node_idx(node_id);
    if (idx >= 0) {
        return &g_nodes.nodes[idx];
    }
    return NULL;
}

node_t* node_manager_get_by_num(uint32_t node_num) {
    for (int i = 0; i < MAX_NODES; i++) {
        if (g_nodes.nodes[i].is_valid && 
            g_nodes.nodes[i].node_num == node_num) {
            return &g_nodes.nodes[i];
        }
    }
    return NULL;
}

int node_manager_get_all(node_t *nodes, int max_count) {
    int count = 0;
    for (int i = 0; i < MAX_NODES && count < max_count; i++) {
        if (g_nodes.nodes[i].is_valid) {
            memcpy(&nodes[count], &g_nodes.nodes[i], sizeof(node_t));
            count++;
        }
    }
    return count;
}

int node_manager_remove(const char *node_id) {
    int idx = find_node_idx(node_id);
    if (idx >= 0) {
        g_nodes.nodes[idx].is_valid = false;
        g_nodes.count--;
        return 0;
    }
    return -1;
}

void node_manager_check_expired(void) {
    uint64_t now = get_time_ms();
    
    for (int i = 0; i < MAX_NODES; i++) {
        if (g_nodes.nodes[i].is_valid) {
            uint64_t age = (now - g_nodes.nodes[i].last_heard) / 1000;
            if (age > g_nodes.expire_time) {
                log_info("node", "Node %s expired", g_nodes.nodes[i].node_id);
                g_nodes.nodes[i].is_valid = false;
                g_nodes.count--;
            }
        }
    }
}

int node_manager_get_count(void) {
    return g_nodes.count;
}
```
